103 async fn resolve_wasi_module_bytes(workload: &ComputeWorkload) -> Result<Vec<u8>> {
104 match workload {
105 ComputeWorkload::WasiPreview1 {
106 module_hex: Some(module_hex),
107 module_ref: None,
108 ..
109 } => {
110 let module = decode_hex_limited(module_hex, MAX_WASI_MODULE_BYTES, "wasi module")?;
111 validate_wasm_module_bytes(&module, MAX_WASI_MODULE_BYTES)?;
112 Ok(module)
113 }
114 ComputeWorkload::WasiPreview1 {
115 module_hex: None,
116 module_ref: Some(module_ref),
117 ..
118 } => fetch_wasi_module_ref(module_ref).await,
119 ComputeWorkload::WasiPreview1 { .. } => {
120 bail!("wasi workload requires exactly one module source")
121 }
122 _ => bail!("compute workload is not wasi_preview1"),
123 }

126 async fn fetch_wasi_module_ref(module_ref: &ComputeModuleRef) -> Result<Vec<u8>> {
127 let client = reqwest::Client::builder()
128 .timeout(Duration::from_secs(MODULE_FETCH_TIMEOUT_SECS))
129 .build()?;
130 let manifest_url = format!(
131 "{}/v1/storage/contracts/{}/manifest",
132 module_ref.host_url.trim_end_matches('/'),
133 module_ref.contract_id
134 );
135 let manifest: StorageBundleManifest = client
136 .get(manifest_url)
137 .send()
138 .await?
139 .error_for_status()?
140 .json()
141 .await?;
142 let file = validate_module_manifest(module_ref, &manifest)?;
143 let file_size_bytes = file.size_bytes;
144 let module_url = format!(
145 "{}/public/{}/{}",
146 module_ref.host_url.trim_end_matches('/'),
147 module_ref.contract_id,
148 module_ref.path
149 );
150 let bytes = client
151 .get(module_url)
152 .send()
153 .await?
154 .error_for_status()?
155 .bytes()
156 .await?;
157 if bytes.len() as u64 != module_ref.size_bytes || bytes.len() as u64 != file_size_bytes {
158 bail!("compute module_ref fetched size mismatch");
159 }
160 if sha256_hex(&bytes) != module_ref.sha256 {
161 bail!("compute module_ref fetched sha256 mismatch");
162 }
163 let module = bytes.to_vec();
164 validate_wasm_module_bytes(&module, MAX_WASI_MODULE_REF_BYTES as usize)?;
165 Ok(module)
166 }

168 fn validate_module_manifest<'a>(
169 module_ref: &ComputeModuleRef,
170 manifest: &'a StorageBundleManifest,
171 ) -> Result<&'a StorageFileEntry> {
172 if manifest.contract_id != module_ref.contract_id {
173 bail!("compute module_ref manifest contract_id mismatch");
174 }
175 if manifest.merkle_root != module_ref.merkle_root {
176 bail!("compute module_ref manifest merkle_root mismatch");
177 }
178 if !matches!(manifest.mode, StorageMode::PublicRaw { .. }) {
179 bail!("compute module_ref storage contract must be public_raw");
180 }
181 let file = manifest
182 .files
183 .iter()
184 .find(|file| file.path == module_ref.path)
185 .ok_or_else(|| anyhow!("compute module_ref path is missing from storage manifest"))?;
186 if file.size_bytes != module_ref.size_bytes {
187 bail!("compute module_ref manifest size mismatch");
188 }
189 if file.sha256 != module_ref.sha256 {
190 bail!("compute module_ref manifest sha256 mismatch");
191 }
192 Ok(file)
193 }

289 fn run_wasi_shard(
290 spec: &ComputeJobSpec,
291 module_bytes: &[u8],
292 shard: &ComputeShardSpec,
293 tx_hash: &str,
294 provider: &str,
295 scratch_dir: &Path,
296 artifact_root: &Path,
297 ) -> std::result::Result<WasiRunSuccess, WasiRunFailure> {
298 let (args, env) = match &spec.workload {
299 ComputeWorkload::WasiPreview1 { args, env, .. } => (args, env),
300 _ => return Err(failure("compute workload is not wasi_preview1")),
301 };
302 let ComputeShardInput::Wasi { stdin_hex, files } = &shard.input else {
303 return Err(failure("compute shard input is not wasi"));
304 };
305
306 if let Err(error) = materialize_inputs(stdin_hex.as_deref(), files, scratch_dir) {
307 return Err(failure(error.to_string()));
308 }
309
310 let stdout = MemoryOutputPipe::new(spec.sandbox.max_stdout_bytes);
311 let stderr = MemoryOutputPipe::new(spec.sandbox.max_stderr_bytes);
312 let stdout_reader = stdout.clone();
313 let stderr_reader = stderr.clone();
314
315 let run_result = run_wasi_module(
316 module_bytes,
317 args,
318 env,
319 stdin_hex.as_deref(),
320 scratch_dir,
321 stdout,
322 stderr,
323 spec,
324 );
325
326 let stdout = stdout_reader.contents().to_vec();
327 let stderr = stderr_reader.contents().to_vec();
328
329 if let Err(error) = run_result {
330 let artifacts =
331 collect_declared_artifacts(spec, shard, tx_hash, provider, scratch_dir, artifact_root)
332 .unwrap_or_default();
333 return Err(WasiRunFailure {
334 message: normalize_sandbox_error(&error),
335 artifacts,
336 stdout,
337 stderr,
338 });
339 }
340
341 let artifacts = match collect_declared_artifacts(
342 spec,
343 shard,
344 tx_hash,
345 provider,
346 scratch_dir,
347 artifact_root,
348 ) {
349 Ok(artifacts) => artifacts,
350 Err(error) => {
351 return Err(WasiRunFailure {
352 message: error.to_string(),
353 artifacts: Vec::new(),
354 stdout,
355 stderr,
356 });
357 }
358 };
359 let output = match read_wasi_output_json(scratch_dir) {
360 Ok(output) => output,
361 Err(error) => {
362 return Err(WasiRunFailure {
363 message: error.to_string(),
364 artifacts,
365 stdout,
366 stderr,
367 });
368 }
369 };
370 Ok(WasiRunSuccess {
371 output,
372 artifacts,
373 stdout,
374 stderr,
375 })
376 }

378 #[allow(clippy::too_many_arguments)]
379 fn run_wasi_module(
380 module_bytes: &[u8],
381 args: &[String],
382 env: &BTreeMap<String, String>,
383 stdin_hex: Option<&str>,
384 scratch_dir: &Path,
385 stdout: MemoryOutputPipe,
386 stderr: MemoryOutputPipe,
387 spec: &ComputeJobSpec,
388 ) -> Result<()> {
389 let mut config = Config::new();
390 config.consume_fuel(true);
391 config.epoch_interruption(true);
392 let engine = Engine::new(&config)?;
393 let module = Module::from_binary(&engine, module_bytes)?;
394 let mut linker: Linker<SandboxState> = Linker::new(&engine);
395 p1::add_to_linker_sync(&mut linker, |state| &mut state.wasi)?;
396
397 let stdin = match stdin_hex {
398 Some(input) => decode_hex_limited(
399 input,
400 crate::compute::MAX_WASI_INPUT_FILE_BYTES,
401 "wasi stdin",
402 )?,
403 None => Vec::new(),
404 };
405 let mut argv = Vec::with_capacity(args.len() + 1);
406 argv.push("job.wasm".to_string());
407 argv.extend(args.iter().cloned());
408 let env = env
409 .iter()
410 .map(|(key, value)| (key.clone(), value.clone()))
411 .collect::<Vec<_>>();
412
413 let mut wasi = WasiCtxBuilder::new();
414 wasi.stdin(MemoryInputPipe::new(stdin))
415 .stdout(stdout)
416 .stderr(stderr)
417 .args(&argv)
418 .envs(&env)
419 .allow_blocking_current_thread(true);
420 wasi.preopened_dir(
421 scratch_dir,
422 WASI_WORK_DIR,
423 DirPerms::all(),
424 FilePerms::all(),
425 )?;
426
427 let limits = StoreLimitsBuilder::new()
428 .memory_size(spec.sandbox.max_memory_bytes as usize)
429 .instances(1)
430 .tables(4)
431 .memories(1)
432 .trap_on_grow_failure(true)
433 .build();
434 let mut store = Store::new(
435 &engine,
436 SandboxState {
437 wasi: wasi.build_p1(),
438 limits,
439 },
440 );
441 store.limiter(|state| &mut state.limits);
442 store.set_fuel(spec.sandbox.max_fuel)?;
443 store.set_epoch_deadline(1);
444 store.epoch_deadline_trap();
445
446 let instance = linker.instantiate(&mut store, &module)?;
447 let start = instance.get_typed_func::<(), ()>(&mut store, "_start")?;
448 let _epoch_deadline = start_epoch_deadline(engine.clone(), spec.max_runtime_secs);
449 start.call(&mut store, ())?;
450 Ok(())
451 }

453 fn start_epoch_deadline(engine: Engine, max_runtime_secs: u64) -> EpochDeadlineGuard {
454 let (done_tx, done_rx) = mpsc::channel();
455 let timeout = Duration::from_secs(max_runtime_secs);
456 let handle = thread::spawn(move || {
457 if done_rx.recv_timeout(timeout).is_err() {
458 engine.increment_epoch();
459 }
460 });
461 EpochDeadlineGuard {
462 done: Some(done_tx),
463 handle: Some(handle),
464 }
465 }

502 fn collect_declared_artifacts(
503 spec: &ComputeJobSpec,
504 shard: &ComputeShardSpec,
505 tx_hash: &str,
506 provider: &str,
507 scratch_dir: &Path,
508 artifact_root: &Path,
509 ) -> Result<Vec<ComputeArtifactRef>> {
510 let mut artifacts = Vec::new();
511 let mut total = 0u64;
512 for artifact in &spec.artifact_policy.outputs {
513 validate_compute_artifact_path(&artifact.path)?;
514 let source = scratch_dir.join(&artifact.path);
515 if !source.exists() {
516 if artifact.required {
517 bail!(
518 "required compute artifact {} was not created",
519 artifact.path
520 );
521 }
522 continue;
523 }
524 let metadata = fs::symlink_metadata(&source)?;
525 if !metadata.is_file() {
526 bail!("compute artifact {} is not a regular file", artifact.path);
527 }
528 let size = metadata.len();
529 let max_bytes = artifact
530 .max_bytes
531 .unwrap_or(spec.artifact_policy.max_total_bytes);
532 if size > max_bytes {
533 bail!(
534 "compute artifact {} exceeds its declared size",
535 artifact.path
536 );
537 }
538 total = total
539 .checked_add(size)
540 .ok_or_else(|| anyhow!("compute artifact total size overflow"))?;
541 if total > spec.artifact_policy.max_total_bytes {
542 bail!("compute artifacts exceed max_total_bytes");
543 }
544 let bytes = fs::read(&source)?;
545 let sha256 = sha256_hex(&bytes);
546 let archive_path = artifact_root
547 .join(tx_hash)
548 .join(&shard.shard_id)
549 .join(&artifact.path);
550 write_sandbox_file(&archive_path, &bytes)?;
551 artifacts.push(artifact_ref(
552 artifact,
553 tx_hash,
554 &shard.shard_id,
555 provider,
556 size,
557 sha256,
558 ));
559 }
560 Ok(artifacts)
561 }

609 fn normalize_sandbox_error(error: &anyhow::Error) -> String {
610 let message = error.to_string();
611 let detail = format!("{error:?}");
612 if detail.contains("all fuel consumed") {
613 "compute sandbox exhausted fuel".to_string()
614 } else if detail.contains("interrupt") || detail.contains("epoch") {
615 "compute sandbox exceeded runtime limit".to_string()
616 } else if detail.contains("write beyond capacity of MemoryOutputPipe") {
617 "compute sandbox exceeded stdio limit".to_string()
618 } else if detail.contains("wasm trap") {
619 "compute sandbox trapped".to_string()
620 } else {
621 message
622 }
623 }
