181 pub async fn build_storage_bundle(
182 input: &Path,
183 store_dir: &Path,
184 options: StorageBundleBuildOptions,
185 ) -> Result<StorageBundleBuild> {
186 validate_contract_id_for_path(&options.contract_id)?;
187 if options.chunk_size_bytes == 0 || options.chunk_size_bytes > usize::MAX as u64 {
188 bail!("storage chunk size is invalid for this platform");
189 }
190
191 let contract_dir = contract_dir(store_dir, &options.contract_id)?;
192 let chunks_dir = contract_dir.join(CHUNKS_DIR);
193 if fs::try_exists(contract_dir.join(BUNDLE_MANIFEST_FILE)).await? {
194 bail!(
195 "storage bundle {} already exists in {}",
196 options.contract_id,
197 contract_dir.display()
198 );
199 }
200 fs::create_dir_all(&chunks_dir)
201 .await
202 .with_context(|| format!("failed to create {}", chunks_dir.display()))?;
203
204 let source_files = collect_source_files(input)?;
205 if source_files.is_empty() {
206 bail!("storage bundle input is empty");
207 }
208
209 let index_path = options.index_path.clone().unwrap_or_else(|| {
210 if source_files.len() == 1 {
211 source_files[0].storage_path.clone()
212 } else {
213 "index.html".to_string()
214 }
215 });
216
217 let encryption = match options.mode {
218 StorageBundleMode::Encrypted => Some(random_chunk_encryption()),
219 StorageBundleMode::PublicRaw => None,
220 };
221 let mut writer = ChunkWriter::new(
222 chunks_dir.clone(),
223 options.chunk_size_bytes as usize,
224 encryption.clone(),
225 );
226 let mut files = Vec::new();
227 let mut critical_prefix_hash = None;
228 let mut critical_prefix_bytes = None;
229
230 for source in source_files {
231 let bytes = fs::read(&source.disk_path)
232 .await
233 .with_context(|| format!("failed to read {}", source.disk_path.display()))?;
234 let offset = writer.source_size_bytes;
235 if options.mode == StorageBundleMode::PublicRaw && source.storage_path == index_path {
236 let prefix_len = bytes
237 .len()
238 .min(STATIC_SITE_CRITICAL_PREFIX_TARGET_BYTES as usize);
239 if prefix_len > 0 {
240 critical_prefix_hash = Some(sha256_hex(&bytes[..prefix_len]));
241 critical_prefix_bytes = Some(prefix_len as u32);
242 }
243 }
244 writer.push(&bytes).await?;
245 files.push(StorageFileEntry {
246 path: source.storage_path.clone(),
247 offset,
248 size_bytes: bytes.len() as u64,
249 sha256: sha256_hex(&bytes),
250 content_type: content_type_for_path(&source.storage_path).to_string(),
251 });
252 }
253
254 let chunks = writer.finish().await?;
255 let source_size_bytes = writer.source_size_bytes;
256 let size_bytes = writer.size_bytes;
257 if chunks.is_empty() || size_bytes == 0 {
258 bail!("storage bundle input produced no chunks");
259 }
260 let chunk_hashes = chunks
261 .iter()
262 .map(|chunk| chunk.sha256.clone())
263 .collect::<Vec<_>>();
264 let merkle_root = merkle_root_from_chunk_hashes(&chunk_hashes)?;
265
266 let mode = match options.mode {
267 StorageBundleMode::Encrypted => StorageMode::Encrypted,
268 StorageBundleMode::PublicRaw => StorageMode::PublicRaw {
269 manifest: Some(StaticSiteManifest {
270 index_path,
271 critical_prefix_hash,
272 critical_prefix_bytes,
273 compression: StorageCompression::Identity,
274 content_types: files
275 .iter()
276 .map(|file| (file.path.clone(), file.content_type.clone()))
277 .collect::<BTreeMap<_, _>>(),
278 }),
279 },
280 };
281 let encryption_info = encryption.as_ref().map(|value| StorageEncryptionInfo {
282 cipher: "xchacha20poly1305".to_string(),
283 nonce_prefix_hex: hex::encode(value.nonce_prefix),
284 });
285
286 let spec = StorageContractSpec {
287 contract_id: options.contract_id.clone(),
288 host: options.host,
289 mode: mode.clone(),
290 size_bytes,
291 chunk_size_bytes: options.chunk_size_bytes,
292 merkle_root: merkle_root.clone(),
293 duration_secs: options.duration_secs,
294 proof_interval_secs: options.proof_interval_secs,
295 proof_sample_count: options.proof_sample_count,
296 reward_rate_per_64mib_second: options.reward_rate_per_64mib_second,
297 };
298 validate_storage_contract_spec(&spec)?;
299
300 let manifest = StorageBundleManifest {
301 version: STORAGE_BUNDLE_VERSION,
302 contract_id: options.contract_id.clone(),
303 mode,
304 chunk_size_bytes: options.chunk_size_bytes,
305 source_size_bytes,
306 size_bytes,
307 merkle_root,
308 files,
309 chunks,
310 encryption: encryption_info,
311 };
312
313 let manifest_path = contract_dir.join(BUNDLE_MANIFEST_FILE);
314 let spec_path = contract_dir.join(CONTRACT_SPEC_FILE);
315 write_json_file(&manifest_path, &manifest).await?;
316 write_json_file(&spec_path, &spec).await?;
317
318 let secrets_path = if let Some(encryption) = encryption {
319 let secrets = StorageEncryptionSecrets {
320 cipher: "xchacha20poly1305".to_string(),
321 key_hex: hex::encode(encryption.key),
322 nonce_prefix_hex: hex::encode(encryption.nonce_prefix),
323 };
324 let path = contract_dir.join(ENCRYPTION_SECRETS_FILE);
325 write_json_file(&path, &secrets).await?;
326 Some(path)
327 } else {
328 None
329 };
330
331 Ok(StorageBundleBuild {
332 spec,
333 manifest,
334 contract_dir,
335 manifest_path,
336 spec_path,
337 secrets_path,
338 })
339 }

393 pub async fn storage_sample_from_manifest(
394 store_dir: &Path,
395 manifest: &StorageBundleManifest,
396 chunk_index: u64,
397 ) -> Result<StorageProofSample> {
398 let chunk = manifest
399 .chunks
400 .get(chunk_index as usize)
401 .ok_or_else(|| anyhow!("unknown storage chunk index {chunk_index}"))?;
402 if chunk.index != chunk_index {
403 bail!("storage manifest chunk index mismatch");
404 }
405 let bytes = read_chunk_bytes(store_dir, &manifest.contract_id, chunk_index).await?;
406 let actual_hash = sha256_hex(&bytes);
407 if actual_hash != chunk.sha256 {
408 bail!("storage chunk hash mismatch for index {chunk_index}");
409 }
410 let hashes = manifest
411 .chunks
412 .iter()
413 .map(|entry| entry.sha256.clone())
414 .collect::<Vec<_>>();
415 let proof = merkle_proof_from_chunk_hashes(&hashes, chunk_index)?;
416 let sample = StorageProofSample {
417 chunk_index,
418 chunk_hash: chunk.sha256.clone(),
419 proof,
420 };
421 validate_storage_proof_sample(&sample, &manifest.merkle_root)?;
422 Ok(sample)

464 pub fn merkle_root_from_chunk_hashes(hashes: &[String]) -> Result<String> {
465 let layers = merkle_layers(hashes)?;
466 let root = layers
467 .last()
468 .and_then(|layer| layer.first())
469 .ok_or_else(|| anyhow!("storage merkle tree requires at least one chunk"))?;
470 Ok(hex::encode(root))
471 }
472
473 pub fn merkle_proof_from_chunk_hashes(
474 hashes: &[String],
475 chunk_index: u64,
476 ) -> Result<Vec<MerkleProofNode>> {
477 let layers = merkle_layers(hashes)?;
478 let mut index = usize::try_from(chunk_index).map_err(|_| anyhow!("chunk index too large"))?;
479 if index >= hashes.len() {
480 bail!("chunk index is outside the merkle tree");
481 }
482
483 let mut proof = Vec::new();
484 for layer in layers.iter().take(layers.len().saturating_sub(1)) {
485 let is_right = index % 2 == 1;
486 let sibling_index = if is_right { index - 1 } else { index + 1 };
487 if let Some(sibling) = layer.get(sibling_index) {
488 proof.push(MerkleProofNode {
489 side: if is_right {
490 MerkleProofSide::Left
491 } else {
492 MerkleProofSide::Right
493 },
494 hash: hex::encode(sibling),
495 });
496 }
497 index /= 2;
498 }
499 Ok(proof)
500 }

626 async fn read_public_file_bytes(
627 store_dir: &Path,
628 contract_id: &str,
629 request_path: &str,
630 ) -> Result<(Vec<u8>, String)> {
631 let manifest = load_bundle_manifest(store_dir, contract_id).await?;
632 let public_manifest = match &manifest.mode {
633 StorageMode::PublicRaw { manifest } => manifest
634 .as_ref()
635 .ok_or_else(|| anyhow!("public storage bundle is missing its static manifest"))?,
636 StorageMode::Encrypted => bail!("encrypted storage bundles cannot be served publicly"),
637 };
638 let request_path = normalize_public_request_path(request_path)?;
639 let target_path = if request_path.is_empty() {
640 public_manifest.index_path.as_str()
641 } else {
642 request_path.as_str()
643 };
644 let file = manifest
645 .files
646 .iter()
647 .find(|file| file.path == target_path)
648 .or_else(|| {
649 if request_path.contains('.') {
650 None
651 } else {
652 manifest
653 .files
654 .iter()
655 .find(|file| file.path == public_manifest.index_path)
656 }
657 })
658 .ok_or_else(|| anyhow!("public file not found"))?;
659
660 let mut remaining = file.size_bytes as usize;
661 let mut offset = file.offset;
662 let mut out = Vec::with_capacity(remaining);
663 while remaining > 0 {
664 let chunk_index = offset / manifest.chunk_size_bytes;
665 let chunk_offset = (offset % manifest.chunk_size_bytes) as usize;
666 let chunk = read_chunk_bytes(store_dir, &manifest.contract_id, chunk_index).await?;
667 if chunk_offset >= chunk.len() {
668 bail!("public file offset is outside chunk bounds");
669 }
670 let take = remaining.min(chunk.len() - chunk_offset);
671 out.extend_from_slice(&chunk[chunk_offset..chunk_offset + take]);
672 remaining -= take;
673 offset += take as u64;
674 }
675 if sha256_hex(&out) != file.sha256 {
676 bail!("public file hash mismatch");
677 }
678 Ok((out, file.content_type.clone()))
679 }

761 impl ChunkWriter {
762 fn new(
763 chunks_dir: PathBuf,
764 chunk_size_bytes: usize,
765 encryption: Option<ChunkEncryption>,
766 ) -> Self {
767 Self {
768 chunks_dir,
769 chunk_size_bytes,
770 encryption,
771 current: Vec::with_capacity(chunk_size_bytes),
772 chunks: Vec::new(),
773 source_size_bytes: 0,
774 size_bytes: 0,
775 }
776 }
777
778 async fn push(&mut self, mut bytes: &[u8]) -> Result<()> {
779 self.source_size_bytes += bytes.len() as u64;
780 while !bytes.is_empty() {
781 let remaining_in_chunk = self.chunk_size_bytes - self.current.len();
782 let take = remaining_in_chunk.min(bytes.len());
783 self.current.extend_from_slice(&bytes[..take]);
784 bytes = &bytes[take..];
785 if self.current.len() == self.chunk_size_bytes {
786 self.flush_current().await?;
787 }
788 }
789 Ok(())
790 }
791
792 async fn finish(&mut self) -> Result<Vec<StorageChunkEntry>> {
793 if !self.current.is_empty() {
794 self.flush_current().await?;
795 }
796 Ok(self.chunks.clone())
797 }
798
799 async fn flush_current(&mut self) -> Result<()> {
800 let index = self.chunks.len() as u64;
801 let plain = std::mem::take(&mut self.current);
802 let chunk = if let Some(encryption) = &self.encryption {
803 encryption.encrypt_chunk(index, &plain)?
804 } else {
805 plain
806 };
807 let path = self.chunks_dir.join(format!("{index:020}.chunk"));
808 fs::write(&path, &chunk)
809 .await
810 .with_context(|| format!("failed to write {}", path.display()))?;
811 let sha256 = sha256_hex(&chunk);
812 self.size_bytes += chunk.len() as u64;
813 self.chunks.push(StorageChunkEntry {
814 index,
815 size_bytes: chunk.len() as u64,
816 sha256,
817 });
818 self.current = Vec::with_capacity(self.chunk_size_bytes);
819 Ok(())
820 }
821 }
822
823 impl ChunkEncryption {
824 fn encrypt_chunk(&self, index: u64, bytes: &[u8]) -> Result<Vec<u8>> {
825 let cipher = XChaCha20Poly1305::new_from_slice(&self.key)
826 .map_err(|_| anyhow!("invalid storage encryption key length"))?;
827 cipher
828 .encrypt(XNonce::from_slice(&self.nonce(index)), bytes)
829 .map_err(|_| anyhow!("failed to encrypt storage chunk"))
830 }
831
832 fn nonce(&self, index: u64) -> [u8; 24] {
833 let mut nonce = [0u8; 24];
834 nonce[..16].copy_from_slice(&self.nonce_prefix);
835 nonce[16..].copy_from_slice(&index.to_be_bytes());
836 nonce
837 }
838 }
839
840 fn random_chunk_encryption() -> ChunkEncryption {
841 let mut rng = thread_rng();
842 let mut key = [0u8; 32];
843 let mut nonce_prefix = [0u8; 16];
844 rng.fill_bytes(&mut key);
845 rng.fill_bytes(&mut nonce_prefix);
846 ChunkEncryption { key, nonce_prefix }
847 }

849 fn collect_source_files(input: &Path) -> Result<Vec<SourceFile>> {
850 let metadata = std::fs::metadata(input)
851 .with_context(|| format!("failed to inspect storage input {}", input.display()))?;
852 if metadata.is_file() {
853 let file_name = input
854 .file_name()
855 .and_then(|value| value.to_str())
856 .ok_or_else(|| anyhow!("storage input file name must be valid UTF-8"))?;
857 return Ok(vec![SourceFile {
858 disk_path: input.to_path_buf(),
859 storage_path: file_name.to_string(),
860 }]);
861 }
862 if !metadata.is_dir() {
863 bail!("storage input must be a file or directory");
864 }
865
866 let mut stack = vec![input.to_path_buf()];
867 let mut files = Vec::new();
868 while let Some(dir) = stack.pop() {
869 for entry in std::fs::read_dir(&dir)
870 .with_context(|| format!("failed to read directory {}", dir.display()))?
871 {
872 let entry = entry?;
873 let path = entry.path();
874 let file_type = entry.file_type()?;
875 if file_type.is_dir() {
876 stack.push(path);
877 } else if file_type.is_file() {
878 files.push(SourceFile {
879 storage_path: storage_relative_path(input, &path)?,
880 disk_path: path,
881 });
882 }
883 }
884 }
885 files.sort_by(|left, right| left.storage_path.cmp(&right.storage_path));
886 Ok(files)
887 }

889 fn storage_relative_path(root: &Path, path: &Path) -> Result<String> {
890 let relative = path
891 .strip_prefix(root)
892 .with_context(|| format!("{} is not under {}", path.display(), root.display()))?;
893 let mut parts = Vec::new();
894 for component in relative.components() {
895 match component {
896 Component::Normal(part) => {
897 let part = part
898 .to_str()
899 .ok_or_else(|| anyhow!("storage paths must be valid UTF-8"))?;
900 if part.is_empty() || part == "." || part == ".." {
901 bail!("storage path component is invalid");
902 }
903 parts.push(part.to_string());
904 }
905 Component::CurDir => {}
906 _ => bail!("storage paths may not contain parent or absolute components"),
907 }
908 }
909 if parts.is_empty() {
910 bail!("storage path is empty");
911 }
912 Ok(parts.join("/"))
913 }

949 fn merkle_layers(hashes: &[String]) -> Result<Vec<Vec<[u8; 32]>>> {
950 if hashes.is_empty() {
951 bail!("storage merkle tree requires at least one chunk");
952 }
953 let mut layers = vec![
954 hashes
955 .iter()
956 .map(|hash| decode_sha256_hex(hash))
957 .collect::<Result<Vec<_>>>()?,
958 ];
959 while layers.last().map(Vec::len).unwrap_or(0) > 1 {
960 let previous = layers.last().unwrap();
961 let mut next = Vec::with_capacity(previous.len().div_ceil(2));
962 for pair in previous.chunks(2) {
963 if pair.len() == 1 {
964 next.push(pair[0]);
965 } else {
966 let mut hasher = Sha256::new();
967 hasher.update(pair[0]);
968 hasher.update(pair[1]);
969 next.push(hasher.finalize().into());
970 }
971 }
972 layers.push(next);
973 }
974 Ok(layers)
975 }

977 fn decode_sha256_hex(input: &str) -> Result<[u8; 32]> {
978 let bytes = hex::decode(input).context("invalid sha256 hex")?;
979 if bytes.len() != 32 {
980 bail!("sha256 hex must be 32 bytes");
981 }
982 let mut out = [0u8; 32];
983 out.copy_from_slice(&bytes);
984 Ok(out)
985 }
986
987 fn sha256_hex(bytes: &[u8]) -> String {
988 hex::encode(Sha256::digest(bytes))
989 }
