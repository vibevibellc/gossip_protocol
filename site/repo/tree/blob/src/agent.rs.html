83 pub async fn run(self) -> Result<()> {
84 let runtime = Arc::new(self);
85 let app = Router::new()
86 .route(
87 "/v1/internal/agent/confirm",
88 post(run_probe_agent_confirmation),
89 )
90 .route("/v1/internal/agent/browser", post(run_probe_agent_browser))
91 .route(
92 "/v1/internal/agent/compute-shard",
93 post(run_probe_agent_compute_shard),
94 )
95 .route(
96 "/v1/internal/agent/compute-artifacts/:tx_hash/:shard_id/*path",
97 get(get_probe_agent_compute_artifact),
98 )
99 .with_state(runtime.clone());
100
101 let listener = tokio::net::TcpListener::bind(runtime.config.bind_addr).await?;
102 info!(
103 address = %runtime.address(),
104 bind = %runtime.config.bind_addr,
105 region = ?runtime.config.region,
106 network = ?runtime.config.network,
107 "gossip_protocol probe agent listening"
108 );
109 axum::serve(listener, app).await?;
110 Ok(())
111 }

113 fn require_api_auth(&self, headers: &HeaderMap) -> Result<()> {
114 let Some(expected) = self.config.api_token.as_deref() else {
115 return Ok(());
116 };
117 let supplied = headers
118 .get(AUTHORIZATION)
119 .and_then(|value| value.to_str().ok())
120 .and_then(|value| value.strip_prefix("Bearer "))
121 .ok_or_else(|| anyhow!("missing bearer token for probe agent API"))?;
122 if supplied != expected {
123 bail!("invalid bearer token for probe agent API");
124 }
125 Ok(())
126 }

128 async fn execute_confirmation(
129 &self,
130 lease: SignedProbeAgentLease,
131 ) -> Result<AgentConfirmationResponse> {
132 verify_probe_agent_lease(&lease)?;
133 if lease.body.agent_public_key != self.address() {
134 bail!("probe agent lease is not addressed to this agent");
135 }
136 let now = Utc::now();
137 if now > lease.body.expires_at {
138 bail!("probe agent lease expired");
139 }
140
141 let outcome = execute_health_check(&lease.body.spec).await;
142 let delegated_receipt =
143 self.wallet
144 .sign_delegated_probe_receipt(SignedDelegatedProbeReceipt {
145 id: String::new(),
146 body: DelegatedProbeReceiptBody {
147 chain_id: lease.body.chain_id.clone(),
148 monitor_id: lease.body.monitor_id.clone(),
149 slot_key: lease.body.slot_key.clone(),
150 agent_public_key: String::new(),
151 parent_validator: lease.body.parent_validator.clone(),
152 lease_id: Some(lease.id.clone()),
153 request_id: Some(lease.body.request_id.clone()),
154 region: self.config.region.clone(),
155 network: self.config.network.clone(),
156 observed_at: Utc::now(),
157 response_status: outcome.response_status,
158 latency_ms: outcome.latency_ms,
159 success: outcome.success,
160 assertion_results: outcome.assertion_results,
161 response_headers: outcome.response_headers,
162 response_body_sample: outcome.response_body_sample,
163 error: outcome.error,
164 },
165 signature: String::new(),
166 })?;
167 Ok(AgentConfirmationResponse { delegated_receipt })
168 }

360 async fn get_probe_agent_compute_artifact(
361 State(runtime): State<Arc<ProbeAgentRuntime>>,
362 headers: HeaderMap,
363 AxumPath((tx_hash, shard_id, path)): AxumPath<(String, String, String)>,
364 ) -> std::result::Result<Response, (StatusCode, String)> {
365 runtime.require_api_auth(&headers).map_err(auth_error)?;
366 let artifact_path =
367 compute_artifact_path(&runtime.compute_artifact_dir(), &tx_hash, &shard_id, &path)
368 .map_err(internal_error)?;
369 let bytes = tokio::fs::read(&artifact_path).await.map_err(|_| {
370 (
371 StatusCode::NOT_FOUND,
372 "compute artifact not found".to_string(),
373 )
374 })?;
375 let mut headers = HeaderMap::new();
376 headers.insert(
377 CONTENT_TYPE,
378 HeaderValue::from_static("application/octet-stream"),
379 );
380 Ok((headers, Body::from(bytes)).into_response())
381 }
