47 def do_GET(self) -> None:
48 parsed = urlparse(self.path)
49 if parsed.path != "/health":
50 self.send_error(404, "not found")
51 return
52
53 expected_header = self.headers.get(EXPECTED_HEADER_NAME)
54 region = parse_qs(parsed.query).get("region", [None])[0]
55 if expected_header != EXPECTED_HEADER_VALUE:
56 self._send_json(
57 403,
58 {
59 "ready": False,
60 "error": "missing_or_invalid_header",
61 },
62 )

64 if region != EXPECTED_REGION:
65 self._send_json(
66 400,
67 {
68 "ready": False,
69 "error": "missing_or_invalid_region",
70 },
71 )
72 return

74 self._send_json(
75 200,
76 {
77 "ready": True,
78 "service": {
79 "name": "smoke-health",
80 "version": "1.0.0",
81 },
82 "region": region,

90 def _send_json(self, status: int, payload: dict, extra_headers: Dict[str, str] | None = None) -> None:
91 body = json.dumps(payload).encode("utf-8")
92 self.send_response(status)
93 self.send_header("content-type", "application/json")
94 self.send_header("content-length", str(len(body)))
95 if extra_headers:
96 for name, value in extra_headers.items():
97 self.send_header(name, value)
98 self.end_headers()
99 self.wfile.write(body)

127 def run_cmd(
128 args: List[str],
129 cwd: Path,
130 capture_json: bool = False,
131 extra_env: Dict[str, str] | None = None,
132 ) -> dict | str:
133 env = test_env()
134 if extra_env:
135 env.update(extra_env)
136 completed = subprocess.run(
137 args,
138 cwd=cwd,
139 stdout=subprocess.PIPE,
140 stderr=subprocess.PIPE,
141 text=True,
142 env=env,
143 )
144 if completed.returncode != 0:
145 raise RuntimeError(
146 f"command failed ({completed.returncode}): {' '.join(args)}\n"
147 f"stdout:\n{completed.stdout}\n"
148 f"stderr:\n{completed.stderr}"
149 )
150 if capture_json:
151 return json.loads(completed.stdout)
152 return completed.stdout.strip()

155 def wait_for_http(url: str, timeout_seconds: float) -> None:
156 deadline = time.time() + timeout_seconds
157 last_error = None
158 while time.time() < deadline:
159 try:
160 http_get_json(url)
161 return
162 except (URLError, HTTPError, TimeoutError, ConnectionError, json.JSONDecodeError) as error:
163 last_error = error
164 time.sleep(0.2)
165 raise RuntimeError(f"timed out waiting for {url}: {last_error}")

641 liveness_tx_hash = liveness_submitted_tx["hash"]
642 liveness_finalized = fetch_job_until_finalized(node_urls[0], liveness_tx_hash, 20)
643 surviving_node_urls = [node_urls[0], node_urls[2]]
644 surviving_ledgers = verify_convergence(surviving_node_urls)
645 surviving_statuses = [fetch_status(url) for url in surviving_node_urls]
646 surviving_user_accounts = [fetch_account(url, user_address) for url in surviving_node_urls]

669 if not process_alive(processes[1]):

672 restarted_status = wait_for_status_height(
673 node_urls[1],
674 expected_liveness_height,
675 20,
676 )
677 converged_after_restart = verify_convergence(node_urls)
678 restarted_account = fetch_account(node_urls[1], user_address)
679 if restarted_account["balance"] != expected_post_liveness_balance:
