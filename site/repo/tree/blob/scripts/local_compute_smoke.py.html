68 def run_cmd(args: List[str], capture_json: bool = False) -> dict | str:
69 completed = subprocess.run(
70 args,
71 cwd=REPO_ROOT,
72 stdout=subprocess.PIPE,
73 stderr=subprocess.PIPE,
74 text=True,
75 env=test_env(),
76 )
77 if completed.returncode != 0:
78 raise RuntimeError(
79 f"command failed ({completed.returncode}): {' '.join(args)}\n"
80 f"stdout:\n{completed.stdout}\n"
81 f"stderr:\n{completed.stderr}"
82 )
83 if capture_json:
84 return json.loads(completed.stdout)
85 return completed.stdout.strip()

88 def http_get_json(url: str) -> dict:
89 with urlopen(url, timeout=3) as response:

93 def wait_for_http(url: str, timeout_seconds: float) -> None:
94 deadline = time.time() + timeout_seconds
95 last_error = None
96 while time.time() < deadline:
97 try:
98 http_get_json(url)
99 return
100 except (URLError, HTTPError, TimeoutError, ConnectionError, json.JSONDecodeError) as error:
101 last_error = error
102 time.sleep(0.2)
103 raise RuntimeError(f"timed out waiting for {url}: {last_error}")

245 def stop_process(process: subprocess.Popen) -> None:
246 if process.poll() is not None:
247 return
248 try:
249 os.killpg(process.pid, signal.SIGTERM)
250 except ProcessLookupError:
251 return
252 try:
253 process.wait(timeout=5)
254 except subprocess.TimeoutExpired:
255 os.killpg(process.pid, signal.SIGKILL)
256 process.wait(timeout=5)

259 def fetch_job_until_finalized(node_url: str, tx_hash: str, timeout_seconds: float) -> dict:
260 deadline = time.time() + timeout_seconds
261 last_payload = None
262 while time.time() < deadline:
263 payload = http_get_json(f"{node_url}/v1/control/jobs/{tx_hash}")
264 last_payload = payload
265 if payload.get("finalized"):
266 return payload
267 time.sleep(0.5)
268 raise RuntimeError(f"compute job {tx_hash} did not finalize: {last_payload}")

315 def fetch_storage_contract_until_active(
316 node_url: str, contract_id: str, timeout_seconds: float
317 ) -> dict:
318 deadline = time.time() + timeout_seconds
319 last_payload = None
320 while time.time() < deadline:
321 try:
322 payload = http_get_json(f"{node_url}/v1/control/storage/{contract_id}")
323 last_payload = payload
324 contract = payload.get("contract", {})
325 if contract.get("status") == "active":
326 return payload
327 except (URLError, HTTPError, TimeoutError, ConnectionError, json.JSONDecodeError) as error:
328 last_payload = {"error": str(error)}
329 time.sleep(0.5)
330 raise RuntimeError(f"storage contract {contract_id} did not activate: {last_payload}")
