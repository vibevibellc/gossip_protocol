5 function parseArgs(argv) {
6 const args = {};
7 for (let i = 0; i < argv.length; i += 1) {
8 const part = argv[i];
9 if (!part.startsWith("--")) continue;
10 args[part.slice(2)] = argv[i + 1];
11 i += 1;
12 }

16 function classifyError(error) {
17 const text = `${error?.message || error || ""}`.toLowerCase();
18 if (text.includes("timeout")) return "navigation_timeout";
19 if (text.includes("locator") || text.includes("selector") || text.includes("strict mode violation")) {
20 return "selector_missing";
21 }
22 if (text.includes("assert")) return "assertion_failed";
23 if (text.includes("net::") || text.includes("connection") || text.includes("dns")) {
24 return "network_error";
25 }
26 if (text.includes("browser")) return "browser_crash";
27 return "step_failed";
28 }

30 function resolveInputValue(value, secrets) {
31 if (value.kind === "literal") return value.value;
32 if (value.kind === "secret_ref") {
33 const secret = secrets[value.key];
34 if (typeof secret !== "string") {
35 throw new Error(`missing browser secret ref ${value.key}`);
36 }
37 return secret;
38 }
39 throw new Error(`unsupported browser input kind ${value.kind}`);
40 }

42 function resolveLocator(page, target) {
43 switch (target) {
44 case undefined:
45 throw new Error("missing browser selector");
46 default:
47 break;
48 }
49 if ("css" === target.type || target.kind === "css") {
50 return page.locator(target.value);
51 }
52 if ("text" === target.type || target.kind === "text") {
53 return page.getByText(target.value, { exact: false });
54 }
55 if ("label" === target.type || target.kind === "label") {
56 return page.getByLabel(target.value, { exact: false });
57 }
58 if ("placeholder" === target.type || target.kind === "placeholder") {
59 return page.getByPlaceholder(target.value, { exact: false });
60 }
61 if ("test_id" === target.type || target.kind === "test_id") {
62 return page.getByTestId(target.value);
63 }
64 if ("role" === target.type || target.kind === "role") {
65 return page.getByRole(target.role, target.name ? { name: target.name } : {});
66 }
67 throw new Error(`unsupported browser selector ${JSON.stringify(target)}`);
68 }

74 async function run() {
75 const args = parseArgs(process.argv.slice(2));
76 if (!args.input || !args.output) {

80 const input = JSON.parse(await fs.readFile(args.input, "utf8"));

88 let failedStepIndex = null;
89 let finalUrl = input.package.journey.entry_url;

97 async function createContext(contextId) {
98 const runtime = input.package.runtime;
99 const useSessionState =
100 runtime.cache_mode === "session_state" &&
101 typeof input.session_state_path === "string" &&
102 contextId === "default";
103 const maybeStorageState = useSessionState ? await fileExists(input.session_state_path) : false;
104 const context = await browser.newContext({
105 locale: runtime.locale,
106 timezoneId: runtime.timezone,
107 viewport: {
108 width: runtime.viewport_width,
109 height: runtime.viewport_height
110 },
111 colorScheme: runtime.color_scheme === "dark" ? "dark" : "light",
112 serviceWorkers: runtime.block_service_workers ? "block" : "allow",
113 storageState: maybeStorageState ? input.session_state_path : undefined,
114 recordVideo: input.package.artifact_policy.capture_video
115 ? { dir: input.artifact_dir, size: { width: runtime.viewport_width, height: runtime.viewport_height } }
116 : undefined
117 });
118 if (input.package.artifact_policy.capture_trace) {
119 await context.tracing.start({ screenshots: true, snapshots: true });
120 }
121 const page = await context.newPage();
122 page.on("console", msg => {
123 if (msg.type() === "error") {
124 consoleErrors.push(msg.text());
125 }
126 });
127 page.on("pageerror", error => {
128 consoleErrors.push(String(error));
129 });
130 page.on("requestfailed", request => {
131 const failure = request.failure();
132 networkErrors.push(`${request.method()} ${request.url()} ${failure?.errorText || "failed"}`);
133 });
134 contexts.set(contextId, { context, page });
135 return contexts.get(contextId);
136 }

138 async function ensureCurrentContext() {
139 let entry = contexts.get(currentContextId);
140 if (!entry) {
141 entry = await createContext(currentContextId);
142 }
143 return entry;
144 }
145
146 try {
147 await ensureCurrentContext();
148 for (let index = 0; index < input.package.journey.steps.length; index += 1) {
149 const step = input.package.journey.steps[index];
150 const { page, context } = await ensureCurrentContext();
151 const timeout = input.package.journey.per_step_timeout_ms;
152
153 try {
154 switch (step.kind) {
155 case "navigate":
156 await page.goto(step.url, { waitUntil: "domcontentloaded", timeout });
157 break;
158 case "click":
159 await resolveLocator(page, step.target).first().click({ timeout });
160 break;
161 case "fill":
162 await resolveLocator(page, step.target)
163 .first()
164 .fill(resolveInputValue(step.value, input.secrets), { timeout });
165 break;
166 case "press":
167 await page.keyboard.press(step.key);
168 break;
169 case "wait_for_text":
170 await page.getByText(step.text, { exact: false }).waitFor({ state: "visible", timeout });
171 break;
172 case "assert_text":
173 if (!(await page.getByText(step.text, { exact: false }).first().isVisible({ timeout }))) {
174 throw new Error(`assertion failed: text not visible: ${step.text}`);
175 }
176 break;
177 case "assert_url_contains":
178 if (!page.url().includes(step.text)) {
179 throw new Error(`assertion failed: url does not include ${step.text}`);
180 }
181 break;
182 case "capture_screenshot": {
183 const label = step.label || `step-${index}`;
184 const file = path.join(input.artifact_dir, `${label}.png`);
185 await page.screenshot({ path: file, fullPage: true });
186 screenshotPath = file;
187 break;
188 }
189 case "open_fresh_context":
190 currentContextId = step.context_id;
191 await createContext(step.context_id);
192 break;
193 case "close_context": {
194 const entry = contexts.get(step.context_id);
195 if (entry) {
196 const maybeVideo = entry.page.video();
197 if (input.package.artifact_policy.capture_trace) {
198 const candidate = path.join(input.artifact_dir, `trace-${step.context_id}.zip`);
199 await entry.context.tracing.stop({ path: candidate });
200 tracePath = candidate;
201 }
202 await entry.context.close();
203 if (maybeVideo) {
204 videoPath = await maybeVideo.path().catch(() => videoPath);
205 }
206 contexts.delete(step.context_id);
207 if (currentContextId === step.context_id) {
208 currentContextId = "default";
209 }
210 }
211 break;
212 }
213 default:
214 throw new Error(`unsupported browser step ${step.kind}`);
215 }
216 finalUrl = page.url();
217 void context;
218 } catch (error) {
219 failedStepIndex = index;
220 outcomeClass = classifyError(error);
221 if (input.package.artifact_policy.capture_screenshot_on_failure) {
222 const file = path.join(input.artifact_dir, `failure-step-${index}.png`);
223 await page.screenshot({ path: file, fullPage: true }).catch(() => {});
224 screenshotPath = file;
225 }
226 throw error;
227 }
228 }
229
230 const defaultEntry = contexts.get("default");
231 if (input.package.artifact_policy.capture_trace && defaultEntry) {
232 const candidate = path.join(input.artifact_dir, "trace-default.zip");
233 await defaultEntry.context.tracing.stop({ path: candidate });
234 tracePath = candidate;
235 }
236 if (
237 input.package.runtime.cache_mode === "session_state" &&
238 input.package.session_cache.enabled &&
239 typeof input.session_state_path === "string" &&
240 defaultEntry
241 ) {
242 await defaultEntry.context.storageState({ path: input.session_state_path });
243 }
244 for (const [, entry] of contexts) {
245 const maybeVideo = entry.page.video();
246 await entry.context.close();
247 if (!videoPath && maybeVideo) {
248 videoPath = await maybeVideo.path().catch(() => videoPath);
249 }
250 }
251 await browser.close();
252 await writeOutput(args.output, {
253 success: true,
254 latency_ms: Date.now() - start,
255 failed_step_index: null,
256 final_url: finalUrl,
257 outcome_class: "success",
258 console_error_count: consoleErrors.length,
259 network_error_count: networkErrors.length,
260 screenshot_path: screenshotPath,
261 trace_path: tracePath,
262 video_path: videoPath,
263 error: null
264 });
265 } catch (error) {
266 for (const [, entry] of contexts) {
267 try {
268 const maybeVideo = entry.page.video();
269 if (input.package.artifact_policy.capture_trace) {
270 const candidate = path.join(input.artifact_dir, `trace-${currentContextId}.zip`);
271 await entry.context.tracing.stop({ path: candidate }).catch(() => {});
272 if (!tracePath) tracePath = candidate;
273 }
274 await entry.context.close().catch(() => {});
275 if (!videoPath && maybeVideo) {
276 videoPath = await maybeVideo.path().catch(() => videoPath);
277 }
278 } catch {
279 // ignore cleanup errors
280 }
281 }
282 await browser.close().catch(() => {});
283 await writeOutput(args.output, {
284 success: false,
285 latency_ms: Date.now() - start,
286 failed_step_index: failedStepIndex,
287 final_url: finalUrl,
288 outcome_class: outcomeClass,
289 console_error_count: consoleErrors.length,
290 network_error_count: networkErrors.length,
291 screenshot_path: screenshotPath,
292 trace_path: tracePath,
293 video_path: videoPath,
294 error: String(error?.message || error)
295 });
296 process.exitCode = 1;
297 }
