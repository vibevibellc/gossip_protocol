96 pub async fn quote(&self, request: &SwapQuoteRequest) -> Result<SwapQuote> {
97 validate_swap_quote_request(request)?;
98 let adapter = if let Some(adapter_id) = &request.adapter {
99 self.adapters
100 .iter()
101 .find(|adapter| adapter.id() == adapter_id)
102 .ok_or_else(|| anyhow::anyhow!("unknown adapter {adapter_id}"))?
103 } else {
104 self.adapters
105 .iter()
106 .find(|adapter| adapter.settlement_asset() == &request.settlement_asset)
107 .ok_or_else(|| {
108 anyhow::anyhow!("no adapter available for {:?}", request.settlement_asset)
109 })?
110 };
111
112 adapter.quote(request).await
113 }

179 async fn quote(&self, request: &SwapQuoteRequest) -> Result<SwapQuote> {
180 let settlement_amount = request
181 .token_amount
182 .checked_mul(self.micro_asset_per_token)
183 .and_then(|value| value.checked_div(MICRO_HT))
184 .ok_or_else(|| anyhow::anyhow!("swap amount overflow"))?;
185 Ok(SwapQuote {
186 chain_id: String::new(),
187 quote_id: new_request_id(),
188 wallet: request.wallet.clone(),
189 adapter: self.id.clone(),
190 side: request.side.clone(),
191 settlement_asset: request.settlement_asset.clone(),
192 token_amount: request.token_amount,
193 settlement_amount,
194 settlement_decimals: self.settlement_decimals,
195 expires_at: Utc::now() + Duration::seconds(request.ttl_secs as i64),
196 notes: vec![
197 "This adapter does not touch USDC/USDT rails directly.".into(),
198 "Use the execution plan as a settlement hand-off to your own desk, bridge, or exchange flow.".into(),
199 ],
200 })
201 }

203 async fn execution_plan(&self, quote: &SignedSwapQuote) -> Result<SwapExecutionPlan> {
204 let direction = match quote.quote.side {
205 SwapSide::Buy => "buy HT with the settlement asset",
206 SwapSide::Sell => "sell HT for the settlement asset",
207 };
208 let mut steps = vec![format!(
209 "Use adapter {} to {direction} without embedding any external stablecoin protocol client.",
210 self.id
211 )];
212 if quote.quote.side == SwapSide::Sell {
213 steps.push(
214 "Lock HT on-chain against the signed quote before releasing the off-chain settlement leg."
215 .into(),
216 );
217 }
218 steps.extend(self.instructions.clone());
219 Ok(SwapExecutionPlan {
220 quote: quote.clone(),
221 steps,
222 follow_up_transfer_hint: Some(if quote.quote.side == SwapSide::Sell {
223 "After the settlement leg clears, have the treasury wallet settle the locked HT or let the seller cancel after expiry."
224 .into()
225 } else {
226 "After off-chain settlement, submit a normal HT transfer through the node control API."
227 .into()
228 }),
229 })
230 }

253 async fn quote(&self, request: &SwapQuoteRequest) -> Result<SwapQuote> {
254 let response = self
255 .client
256 .post(&self.quote_url)
257 .headers(self.headers()?)
258 .json(request)
259 .send()
260 .await?
261 .error_for_status()?;
262 Ok(response.json().await?)
263 }
264
265 async fn execution_plan(&self, quote: &SignedSwapQuote) -> Result<SwapExecutionPlan> {
266 let response = self
267 .client
268 .post(&self.execution_url)
269 .headers(self.headers()?)
270 .json(quote)
271 .send()
272 .await?
273 .error_for_status()?;
274
275 let payload: Value = response.json().await?;
276 let steps = payload
277 .get("steps")
278 .and_then(Value::as_array)
279 .map(|items| {
280 items
281 .iter()
282 .filter_map(Value::as_str)
283 .map(ToString::to_string)
284 .collect::<Vec<_>>()
285 })
286 .unwrap_or_else(|| {
287 vec![
288 "External settlement bridge returned no explicit steps.".into(),
289 "Inspect the attached payload and perform the HT transfer once settlement clears."
290 .into(),
291 ]
292 });
293 let follow_up_transfer_hint = payload
294 .get("follow_up_transfer_hint")
295 .and_then(Value::as_str)
296 .map(ToString::to_string);
297
298 Ok(SwapExecutionPlan {
299 quote: quote.clone(),
300 steps,
301 follow_up_transfer_hint,
302 })
303 }
