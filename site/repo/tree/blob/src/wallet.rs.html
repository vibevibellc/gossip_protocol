277 fn from_wallet_file(file: &WalletFile, passphrase: Option<&str>) -> Result<Self> {
278 let bytes = match (&file.keypair_base58, &file.encrypted_keypair) {
279 (Some(keypair_base58), None) => b58decode(keypair_base58)
280 .into_vec()
281 .context("failed to decode wallet keypair")?,
282 (None, Some(encrypted_keypair)) => {
283 let passphrase =
284 passphrase.ok_or_else(|| anyhow::anyhow!("wallet passphrase is required"))?;
285 decrypt_keypair(encrypted_keypair, passphrase)?
286 }
287 (Some(_), Some(_)) => {
288 bail!("wallet file cannot contain both plaintext and encrypted keys")
289 }
290 (None, None) => bail!("wallet file is missing key material"),
291 };
292 if bytes.len() != 64 {
293 bail!("wallet keypair must contain 64 bytes");
294 }
295
296 let secret: [u8; 32] = bytes[..32]
297 .try_into()
298 .map_err(|_| anyhow::anyhow!("invalid secret key length"))?;
299 let public: [u8; 32] = bytes[32..]
300 .try_into()
301 .map_err(|_| anyhow::anyhow!("invalid public key length"))?;
302 let signing_key = SigningKey::from_bytes(&secret);
303 if signing_key.verifying_key().to_bytes() != public {
304 bail!("wallet public key does not match secret key");
305 }
306
307 let wallet = Self { signing_key };
308 if wallet.address() != file.address {
309 bail!("wallet address does not match encoded keypair");
310 }
311 Ok(wallet)
312 }

577 fn encrypt_keypair(raw_keypair: &[u8], passphrase: &str) -> Result<EncryptedKeypair> {
578 let mut salt = [0u8; 16];
579 let mut nonce = [0u8; 24];
580 thread_rng().fill_bytes(&mut salt);
581 thread_rng().fill_bytes(&mut nonce);
582
583 let mut key = [0u8; 32];
584 Argon2::default()
585 .hash_password_into(passphrase.as_bytes(), &salt, &mut key)
586 .map_err(|error| anyhow::anyhow!("failed to derive wallet encryption key: {error}"))?;
587 let cipher = XChaCha20Poly1305::new_from_slice(&key)
588 .map_err(|_| anyhow::anyhow!("failed to initialize wallet cipher"))?;
589 let ciphertext = cipher
590 .encrypt(XNonce::from_slice(&nonce), raw_keypair)
591 .map_err(|_| anyhow::anyhow!("failed to encrypt wallet keypair"))?;
592
593 Ok(EncryptedKeypair {
594 kdf: "argon2id".into(),
595 cipher: "xchacha20poly1305".into(),
596 salt_hex: hex::encode(salt),
597 nonce_hex: hex::encode(nonce),
598 ciphertext_hex: hex::encode(ciphertext),
599 })
600 }
601
602 fn decrypt_keypair(payload: &EncryptedKeypair, passphrase: &str) -> Result<Vec<u8>> {
603 if payload.kdf != "argon2id" {
604 bail!("unsupported wallet key derivation function");
605 }
606 if payload.cipher != "xchacha20poly1305" {
607 bail!("unsupported wallet cipher");
608 }
609 let salt = hex::decode(&payload.salt_hex).context("failed to decode wallet salt")?;
610 let nonce = hex::decode(&payload.nonce_hex).context("failed to decode wallet nonce")?;
611 let ciphertext =
612 hex::decode(&payload.ciphertext_hex).context("failed to decode wallet ciphertext")?;
613 if nonce.len() != 24 {
614 bail!("wallet nonce must contain 24 bytes");
615 }
616
617 let mut key = [0u8; 32];
618 Argon2::default()
619 .hash_password_into(passphrase.as_bytes(), &salt, &mut key)
620 .map_err(|error| anyhow::anyhow!("failed to derive wallet encryption key: {error}"))?;
621 let cipher = XChaCha20Poly1305::new_from_slice(&key)
622 .map_err(|_| anyhow::anyhow!("failed to initialize wallet cipher"))?;
623 let plaintext = cipher
624 .decrypt(XNonce::from_slice(&nonce), ciphertext.as_ref())
625 .map_err(|_| anyhow::anyhow!("failed to decrypt wallet keypair"))?;
626 Ok(plaintext)
627 }
