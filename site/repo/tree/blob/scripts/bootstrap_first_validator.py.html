80 def run_cmd(args: list[str], env: dict[str, str], capture_json: bool = False) -> str | dict:
81 completed = subprocess.run(
82 args,
83 cwd=REPO_ROOT,
84 env=env,
85 text=True,
86 stdout=subprocess.PIPE,
87 stderr=subprocess.PIPE,
88 )
89 if completed.returncode != 0:
90 raise RuntimeError(
91 f"command failed ({completed.returncode}): {' '.join(args)}\n"
92 f"stdout:\n{completed.stdout}\n"
93 f"stderr:\n{completed.stderr}"
94 )
95 stdout = completed.stdout.strip()
96 if capture_json:
97 return json.loads(stdout)
98 return stdout

105 def ensure_empty_output_dir(path: Path) -> None:
106 if path.exists():
107 if not path.is_dir():
108 raise RuntimeError(f"{path} exists and is not a directory")
109 if any(path.iterdir()):
110 raise RuntimeError(f"{path} must be empty")
111 else:
112 path.mkdir(parents=True)

115 def build_binary(profile: str, skip_build: bool, env: dict[str, str]) -> Path:
116 if not skip_build:
117 args = ["cargo", "build"]
118 if profile == "release":
119 args.append("--release")
120 subprocess.run(args, cwd=REPO_ROOT, env=env, check=True)
121 binary = REPO_ROOT / "target" / profile / "gossip_protocol"
122 if not binary.exists():
123 raise RuntimeError(f"expected built binary at {binary}")
124 return binary

132 def parse_bind(bind: str) -> tuple[str, str]:
133 if ":" not in bind:
134 raise RuntimeError("--bind must use HOST:PORT format")
135 host, port = bind.rsplit(":", 1)
136 if not host or not port:
137 raise RuntimeError("--bind must use HOST:PORT format")
138 return host, port

392 args = parse_args()
393
394 passphrase_value = os.environ.get(args.wallet_passphrase_env)
395 if not passphrase_value:
396 raise RuntimeError(
397 f"missing required environment variable {args.wallet_passphrase_env}"

399
400 ensure_empty_output_dir(args.output_dir)
401
402 env = os.environ.copy()
403 env[args.wallet_passphrase_env] = passphrase_value
404
405 binary = build_binary(args.build_profile, args.skip_build, env)
406
407 bind_host, bind_port = parse_bind(args.bind)
408 upstream_host = nginx_upstream_host(bind_host)
409
410 bundle_dir = args.output_dir.resolve()
411 bin_dir = bundle_dir / "bin"
412 wallets_dir = bundle_dir / "wallets"
413 config_dir = bundle_dir / "config"
414 state_dir = bundle_dir / "state"
415 logs_dir = bundle_dir / "logs"
416 systemd_dir = bundle_dir / "systemd"
417 nginx_dir = bundle_dir / "nginx"
418
419 for directory in [bin_dir, wallets_dir, config_dir, state_dir, logs_dir, systemd_dir, nginx_dir]:
420 directory.mkdir(parents=True, exist_ok=True)
421
422 shipped_binary = bin_dir / "gossip_protocol"
423 shutil.copy2(binary, shipped_binary)
424 os.chmod(shipped_binary, 0o755)
425
426 validator_wallet = wallets_dir / "validator.json"
427 requester_wallet = wallets_dir / "requester.json"
428 genesis_path = config_dir / "genesis.json"
429 env_path = config_dir / "node.env"
430 run_script = bundle_dir / "run-validator.sh"
431 submit_script = bundle_dir / "submit-health-check.sh"
432 status_script = bundle_dir / "check-status.sh"
433 service_path = systemd_dir / "gossip-protocol-validator.service"
434 nginx_path = nginx_dir / "gossip_protocol.conf"
435 summary_path = bundle_dir / "BOOTSTRAP_SUMMARY.md"

444 args.wallet_passphrase_env,

447 capture_json=True,
448 )

452 "wallet",
453 "create",
454 str(requester_wallet),
455 "--passphrase-env",
456 args.wallet_passphrase_env,
457 ],
458 env,
459 capture_json=True,
460 )
461
462 validator_address = run_cmd(
463 [str(shipped_binary), "wallet", "address", str(validator_wallet)],
464 env,
465 )
466 requester_address = run_cmd(
467 [str(shipped_binary), "wallet", "address", str(requester_wallet)],
468 env,
469 )

471 genesis_args = [
472 str(shipped_binary),
473 "genesis",
474 "create",
475 "--out",
476 str(genesis_path),
477 "--validator-wallet",
478 str(validator_wallet),

482 str(args.block_time_secs),
483 ]
484 if args.chain_id:
485 genesis_args.extend(["--chain-id", args.chain_id])
486 genesis = run_cmd(genesis_args, env, capture_json=True)
487
488 control_api_token = args.control_api_token or secrets.token_hex(32)
489 gossip_api_token = args.gossip_api_token or secrets.token_hex(32)
490
491 write_text(
492 env_path,
493 render_node_env(
494 args.wallet_passphrase_env,
495 passphrase_value,
496 control_api_token,
497 gossip_api_token,
498 ),
499 0o600,
500 )
501 write_text(
502 run_script,
503 render_run_script(
504 env_path,
505 shipped_binary,
506 validator_wallet,
507 genesis_path,
508 state_dir,
509 args.bind,
510 ),
511 0o755,
512 )
513 write_text(status_script, render_status_script(args.public_host, args.bind), 0o755)
514 write_text(
515 submit_script,
