23 function buildNetwork() {
24 const count = width < 700 ? 22 : 38;
25 nodes = Array.from({ length: count }, (_, index) => {
26 const ring = index % 7;
27 return {
28 x: 40 + Math.random() * Math.max(width - 80, 1),
29 y: 70 + Math.random() * Math.max(height - 140, 1),
30 vx: (Math.random() - 0.5) * 0.22,
31 vy: (Math.random() - 0.5) * 0.22,
32 r: 2.2 + (ring % 3),
33 color: palette[ring % palette.length],
34 };
35 });
36 packets = Array.from({ length: Math.max(10, Math.floor(count / 2)) }, () => spawnPacket());
37 }

39 function nearestPair() {
40 const a = nodes[Math.floor(Math.random() * nodes.length)];
41 let b = nodes[Math.floor(Math.random() * nodes.length)];
42 let best = Infinity;
43 for (const candidate of nodes) {
44 if (candidate === a) continue;
45 const dist = Math.hypot(candidate.x - a.x, candidate.y - a.y);
46 if (dist < best && dist > 80) {
47 best = dist;
48 b = candidate;
49 }
50 }
51 return { a, b };
52 }
