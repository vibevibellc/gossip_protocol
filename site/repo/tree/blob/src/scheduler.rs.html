46 impl RoundRobinPlan {
47 pub fn build(
48 domain: RoundRobinDomain,
49 chain_id: &str,
50 epoch_nonce: &str,
51 providers: &[Address],
52 task_keys: &[String],
53 mandatory_replicas: usize,
54 enable_chunk_audits: bool,
55 ) -> Result<Self> {
56 let mut unique_providers = providers.to_vec();
57 unique_providers.sort();
58 unique_providers.dedup();
59 let seed = compute_hash(&(
60 domain,
61 chain_id.to_string(),
62 epoch_nonce.to_string(),
63 unique_providers.clone(),
64 task_keys,
65 ))?;
66
67 if unique_providers.is_empty() || task_keys.is_empty() {
68 return Ok(Self {
69 domain,
70 seed,
71 providers: unique_providers,
72 task_count: task_keys.len(),
73 lease_span: 0,
74 chunk_count: 0,
75 mandatory_replicas: 0,
76 audit_probability_denominator: 0,
77 leases: Vec::new(),
78 });
79 }
80
81 let shuffled = shuffle_providers(&seed, &unique_providers)?;
82 let provider_count = shuffled.len();
83 let mandatory_replicas = mandatory_replicas.max(1).min(provider_count);
84 let lease_span =
85 average_tasks_per_provider(task_keys.len(), provider_count).next_power_of_two();
86 let chunk_count = div_ceil(task_keys.len(), lease_span);
87 let audit_probability_denominator =
88 if enable_chunk_audits && provider_count > mandatory_replicas {
89 provider_count.next_power_of_two()
90 } else {
91 0
92 };
93
94 let mut leases = Vec::new();
95 for chunk_index in 0..chunk_count {
96 let position_start = chunk_index * lease_span;
97 let position_end_exclusive = (position_start + lease_span).min(task_keys.len());
98 for replica_index in 0..mandatory_replicas {
99 leases.push(RoundRobinLease {
100 provider: shuffled[(chunk_index + replica_index) % provider_count].clone(),
101 replica_index,
102 chunk_index,
103 position_start,
104 position_end_exclusive,
105 audit: false,
106 });
107 }
108
109 if audit_probability_denominator > 0
110 && chunk_is_audited(&seed, chunk_index, audit_probability_denominator)?
111 {
112 let audit_provider =
113 shuffled[(chunk_index + mandatory_replicas) % provider_count].clone();
114 if !leases.iter().any(|lease| {
115 lease.chunk_index == chunk_index && lease.provider == audit_provider
116 }) {
117 leases.push(RoundRobinLease {
118 provider: audit_provider,
119 replica_index: mandatory_replicas,
120 chunk_index,
121 position_start,
122 position_end_exclusive,
123 audit: true,
124 });
125 }
126 }
127 }
128
129 Ok(Self {
130 domain,
131 seed,
132 providers: shuffled,
133 task_count: task_keys.len(),
134 lease_span,
135 chunk_count,
136 mandatory_replicas,
137 audit_probability_denominator,
138 leases,
139 })
140 }

142 pub fn assignment_for_task(&self, task_index: usize) -> Option<TaskAssignment> {
143 if self.lease_span == 0 || task_index >= self.task_count {
144 return None;
145 }
146 let chunk_index = task_index / self.lease_span;
147 let mut mandatory_providers = self
148 .leases
149 .iter()
150 .filter(|lease| lease.chunk_index == chunk_index && !lease.audit)
151 .map(|lease| lease.provider.clone())
152 .collect::<Vec<_>>();
153 mandatory_providers.sort();
154 mandatory_providers.dedup();
155 let audit_provider = self
156 .leases
157 .iter()
158 .find(|lease| lease.chunk_index == chunk_index && lease.audit)
159 .map(|lease| lease.provider.clone());
160 Some(TaskAssignment {
161 mandatory_providers,
162 audit_provider,
163 })
164 }

166 pub fn provider_should_execute(&self, provider: &str, task_index: usize) -> bool {
167 self.assignment_for_task(task_index)
168 .map(|assignment| {
169 assignment
170 .mandatory_providers
171 .iter()
172 .any(|assigned| assigned == provider)
173 || assignment
174 .audit_provider
175 .as_ref()
176 .map(|assigned| assigned == provider)
177 .unwrap_or(false)
178 })
179 .unwrap_or(false)
180 }
181 }

183 fn shuffle_providers(seed: &str, providers: &[Address]) -> Result<Vec<Address>> {
184 let mut scored = providers
185 .iter()
186 .map(|provider| {
187 Ok::<_, anyhow::Error>((
188 compute_hash(&(seed.to_string(), provider.clone()))?,
189 provider.clone(),
190 ))
191 })
192 .collect::<Result<Vec<_>>>()?;
193 scored.sort_by(|left, right| left.0.cmp(&right.0).then_with(|| left.1.cmp(&right.1)));
194 Ok(scored.into_iter().map(|(_, provider)| provider).collect())
195 }

197 fn average_tasks_per_provider(task_count: usize, provider_count: usize) -> usize {
198 div_ceil(task_count.max(1), provider_count.max(1)).max(1)
199 }

201 fn chunk_is_audited(seed: &str, chunk_index: usize, denominator: usize) -> Result<bool> {
202 let sample = compute_hash(&(seed.to_string(), "audit".to_string(), chunk_index))?;
203 let sample = u64::from_str_radix(&sample[..16], 16)?;
204 Ok(sample % denominator as u64 == 0)
205 }

207 fn div_ceil(left: usize, right: usize) -> usize {
208 if right == 0 {
209 return 0;
210 }
211 left.div_ceil(right)
212 }
