765 fn wallet_command(command: WalletCommand) -> Result<()> {
766 match command {
767 WalletCommand::Create(args) => {
768 let wallet = Wallet::generate();
769 let file = if args.insecure_plaintext {
770 if args.passphrase_env.is_some() {
771 bail!("--insecure-plaintext cannot be combined with --passphrase-env");
772 }
773 wallet.save_insecure_plaintext(&args.out)?;
774 wallet.to_file_insecure_plaintext()
775 } else {
776 let passphrase = read_required_passphrase(args.passphrase_env.as_deref())?;
777 wallet.save_encrypted(&args.out, &passphrase)?;
778 wallet.to_file_encrypted(&passphrase)?
779 };
780 println!("{}", serde_json::to_string_pretty(&file)?);
781 Ok(())
782 }
783 WalletCommand::Address { wallet } => {
784 let wallet = read_wallet_file(&wallet)?;
785 println!("{}", wallet.address);
786 Ok(())
787 }
788 WalletCommand::Show { wallet } => {
789 let contents = std::fs::read_to_string(&wallet)
790 .with_context(|| format!("failed to read wallet {}", wallet.display()))?;
791 let file: WalletFile = serde_json::from_str(&contents)?;
792 println!("{}", serde_json::to_string_pretty(&file)?);
793 Ok(())
794 }
795 }
796 }

804 async fn create_genesis(args: GenesisCreateArgs) -> Result<()> {
805 let validators = args
806 .validator_wallet
807 .iter()
808 .map(|path| read_wallet_file(path.as_path()))
809 .collect::<Result<Vec<_>>>()?;
810 let validator_addresses = validators
811 .iter()
812 .map(|wallet| wallet.address.clone())
813 .collect::<Vec<_>>();
814 if validator_addresses.is_empty() {
815 bail!("at least one validator wallet is required");
816 }
817
818 let treasury = args
819 .treasury
820 .unwrap_or_else(|| validator_addresses[0].clone());
821 let airdrops = parse_airdrops(&args.airdrop)?;
822 let storage_airdrops = parse_airdrops(&args.storage_airdrop)?;
823 let compute_airdrops = parse_airdrops(&args.compute_airdrop)?;
824 let dns_airdrops = parse_airdrops(&args.dns_airdrop)?;
825 let min_health_receipts =
826 resolve_min_health_receipts(args.min_receipts, validator_addresses.len())?;
827
828 let genesis = GenesisConfig {
829 chain_id: args
830 .chain_id
831 .unwrap_or_else(|| format!("gossip-protocol-{}", new_request_id())),
832 treasury,

1792 fn resolve_storage_window_end(
1793 configured: Option<&str>,
1794 contract: &StorageContractRecord,
1795 ) -> Result<DateTime<Utc>> {
1796 let window_end = if let Some(value) = configured {
1797 DateTime::parse_from_rfc3339(value)
1798 .with_context(|| format!("failed to parse --window-end {value}"))?
1799 .with_timezone(&Utc)
1800 } else {
1801 let mut end = Utc::now();
1802 let max_interval_end =
1803 contract.last_proven_at + Duration::seconds(contract.spec.proof_interval_secs as i64);
1804 if end > max_interval_end {
1805 end = max_interval_end;
1806 }
1807 if end > contract.expires_at {
1808 end = contract.expires_at;
1809 }
1810 end
1811 };
1812
1813 if window_end <= contract.last_proven_at {
1814 bail!("storage proof window_end must be after the contract last_proven_at");
1815 }
1816 if window_end > Utc::now() {
1817 bail!("storage proof window_end cannot be in the future");
1818 }
1819 if window_end > contract.expires_at {
1820 bail!("storage proof window_end cannot exceed contract expiry");
1821 }
1822 let elapsed = window_end
1823 .signed_duration_since(contract.last_proven_at)
1824 .num_seconds();
1825 if elapsed <= 0 || elapsed as u64 > contract.spec.proof_interval_secs {
1826 bail!("storage proof window exceeds the contract proof interval");
1827 }
1828 Ok(window_end)
1829 }
