156 def render_inline_markdown(text: str) -> str:
157 tokens: dict[str, str] = {}
158
159 def token(value: str) -> str:
160 key = f"@@MDTOKEN{len(tokens)}@@"
161 tokens[key] = value
162 return key
163
164 def code_replacement(match: re.Match[str]) -> str:
165 return token(f"<code>{html.escape(match.group(1))}</code>")
166
167 def link_replacement(match: re.Match[str]) -> str:
168 label = html.escape(match.group(1))
169 href = safe_link_href(match.group(2))
170 if href is None:
171 return label
172 escaped_href = html.escape(href, quote=True)
173 return token(f'<a href="{escaped_href}">{label}</a>')
174
175 def bare_url_replacement(match: re.Match[str]) -> str:
176 href = match.group(1).rstrip(".,)")
177 suffix = match.group(1)[len(href) :]
178 escaped_href = html.escape(href, quote=True)
179 escaped_label = html.escape(href)
180 return token(f'<a href="{escaped_href}">{escaped_label}</a>') + suffix
181
182 text = re.sub(r"`([^`]+)`", code_replacement, text)
183 text = re.sub(r"\[([^\]]+)\]\(([^)\s]+)\)", link_replacement, text)
184 text = re.sub(r"(?<![\"'=])\b(https?://[^\s<]+)", bare_url_replacement, text)
185 rendered = html.escape(text)
186 rendered = re.sub(r"\*\*([^*]+)\*\*", r"<strong>\1</strong>", rendered)
187 for key, value in tokens.items():
188 rendered = rendered.replace(key, value)
189 return rendered

192 def split_table_row(line: str) -> list[str]:
193 stripped = line.strip()
194 if stripped.startswith("|"):
195 stripped = stripped[1:]

201 def is_table_separator(line: str) -> bool:
202 cells = split_table_row(line)
203 return bool(cells) and all(TABLE_SEPARATOR_RE.match(cell) for cell in cells)

206 def render_markdown_table(lines: list[str], start: int) -> tuple[str, int] | None:
207 if start + 1 >= len(lines) or not is_table_separator(lines[start + 1]):
208 return None
209 header = split_table_row(lines[start])
210 rows: list[list[str]] = []
211 index = start + 2
212 while index < len(lines) and lines[index].strip().startswith("|"):
213 rows.append(split_table_row(lines[index]))
214 index += 1
215 header_html = "".join(
216 f"<th>{render_inline_markdown(cell)}</th>" for cell in header
217 )
218 body_rows = []
219 for row in rows:
220 padded = row + [""] * max(0, len(header) - len(row))
221 cells = "".join(
222 f"<td>{render_inline_markdown(cell)}</td>" for cell in padded[: len(header)]
223 )
224 body_rows.append(f"<tr>{cells}</tr>")
225 table = f"""<div class="repo-markdown-table-wrap">
226 <table>
227 <thead><tr>{header_html}</tr></thead>
228 <tbody>{''.join(body_rows)}</tbody>
229 </table>
230 </div>"""
231 return table, index

234 def render_markdown_document(text: str) -> str:
235 lines = text.splitlines()
236 output: list[str] = []
237 paragraph: list[str] = []
238 list_tag: str | None = None
239 list_items: list[str] = []
240
241 def flush_paragraph() -> None:
242 nonlocal paragraph
243 if paragraph:
244 joined = " ".join(part.strip() for part in paragraph)
245 output.append(f"<p>{render_inline_markdown(joined)}</p>")
246 paragraph = []
247
248 def flush_list() -> None:
249 nonlocal list_tag, list_items
250 if list_tag and list_items:
251 items = "".join(f"<li>{item}</li>" for item in list_items)
252 output.append(f"<{list_tag}>{items}</{list_tag}>")
253 list_tag = None
254 list_items = []
255
256 index = 0
257 while index < len(lines):
258 line = lines[index]
259 stripped = line.strip()
260
261 if not stripped:
262 flush_paragraph()
263 flush_list()
264 index += 1
265 continue
266
267 if stripped.startswith("```"):
268 flush_paragraph()
269 flush_list()
270 fence_language = stripped.removeprefix("```").strip()
271 code_lines: list[str] = []
272 index += 1
273 while index < len(lines) and not lines[index].strip().startswith("```"):
274 code_lines.append(lines[index])
275 index += 1
276 if index < len(lines):
277 index += 1
278 language_attr = (
279 f' data-language="{html.escape(fence_language, quote=True)}"'
280 if fence_language
281 else ""
282 )
283 output.append(
284 f'<pre class="repo-markdown-code"{language_attr}><code>{html.escape(chr(10).join(code_lines))}</code></pre>'
285 )
286 continue
287
288 table = render_markdown_table(lines, index)
289 if table:
290 flush_paragraph()
291 flush_list()
292 rendered_table, index = table
293 output.append(rendered_table)
294 continue
295
296 heading = re.match(r"^(#{1,6})\s+(.+)$", stripped)
297 if heading:
298 flush_paragraph()
299 flush_list()
300 level = len(heading.group(1))
301 output.append(
302 f"<h{level}>{render_inline_markdown(heading.group(2))}</h{level}>"
303 )
304 index += 1
305 continue
306
307 unordered = re.match(r"^[-*]\s+(.+)$", stripped)
308 ordered = re.match(r"^\d+\.\s+(.+)$", stripped)
309 if unordered or ordered:
310 flush_paragraph()
311 desired_tag = "ul" if unordered else "ol"
312 if list_tag != desired_tag:
313 flush_list()
314 list_tag = desired_tag
315 item_text = (unordered or ordered).group(1)
316 list_items.append(render_inline_markdown(item_text))
317 index += 1
318 continue
319
320 continuation = re.match(r"^\s{2,}(.+)$", line)
321 if continuation and list_tag and list_items:
322 list_items[-1] += " " + render_inline_markdown(continuation.group(1))
323 index += 1
324 continue
325
326 flush_list()
327 paragraph.append(stripped)
328 index += 1
329
330 flush_paragraph()
331 flush_list()
332 return "\n".join(output)

436 try:

440 text = ""

448 rendered_action = ""

450 if is_text:
451 lines = text.splitlines()

464 <table class="repo-code">
