337 pub fn compute_job_cost_for_replication(
338 spec: &ComputeJobSpec,
339 effective_replication: u64,
340 ) -> Result<u64> {
341 validate_compute_job_spec(spec)?;
342 if effective_replication == 0 || effective_replication > u64::from(MAX_COMPUTE_REPLICATION) {
343 bail!("compute effective replication is invalid");
344 }
345 let mut total = COMPUTE_JOB_BASE_COST;
346 total = total
347 .checked_add(spec.shards.len() as u64 * COMPUTE_SHARD_BASE_COST)
348 .ok_or_else(|| anyhow!("compute job cost overflow"))?;
349 if spec.max_runtime_secs > default_compute_runtime_secs() {
350 let extra_secs = spec.max_runtime_secs - default_compute_runtime_secs();
351 let steps = extra_secs.div_ceil(COMPUTE_RUNTIME_SURCHARGE_STEP_SECS);
352 total = total
353 .checked_add(steps * COMPUTE_RUNTIME_SURCHARGE_STEP_COST)
354 .ok_or_else(|| anyhow!("compute job cost overflow"))?;
355 }
356 if matches!(spec.workload, ComputeWorkload::WasiPreview1 { .. }) {
357 total = total
358 .checked_add(COMPUTE_WASI_SURCHARGE_COST)
359 .ok_or_else(|| anyhow!("compute job cost overflow"))?;
360 }
361 if let Some(module_ref) = compute_wasi_module_ref(&spec.workload) {
362 let module_mib = module_ref.size_bytes.div_ceil(1024 * 1024);
363 total = total
364 .checked_add(module_mib * COMPUTE_MODULE_REF_SURCHARGE_PER_MIB)
365 .ok_or_else(|| anyhow!("compute job cost overflow"))?;
366 }
367 if spec.artifact_policy.max_total_bytes > 0 {
368 let artifact_mib = spec.artifact_policy.max_total_bytes.div_ceil(1024 * 1024);
369 total = total
370 .checked_add(artifact_mib * COMPUTE_ARTIFACT_SURCHARGE_PER_MIB)
371 .ok_or_else(|| anyhow!("compute job cost overflow"))?;
372 }
373 total
374 .checked_mul(effective_replication)
375 .ok_or_else(|| anyhow!("compute job cost overflow"))
376 }

378 pub fn validate_compute_job_spec(spec: &ComputeJobSpec) -> Result<()> {
379 if spec.request_id.trim().is_empty() || spec.request_id.len() > MAX_COMPUTE_REQUEST_ID_BYTES {
380 bail!("compute request_id is invalid");
381 }
382 if spec.shards.is_empty() || spec.shards.len() > MAX_COMPUTE_SHARDS {
383 bail!("compute job must include between 1 and {MAX_COMPUTE_SHARDS} shards");
384 }
385 if spec.max_runtime_secs < MIN_COMPUTE_RUNTIME_SECS
386 || spec.max_runtime_secs > MAX_COMPUTE_RUNTIME_SECS
387 {
388 bail!(
389 "compute max_runtime_secs must be between {MIN_COMPUTE_RUNTIME_SECS} and {MAX_COMPUTE_RUNTIME_SECS}"
390 );
391 }
392 if spec.replication == 0 || spec.replication > MAX_COMPUTE_REPLICATION {
393 bail!("compute replication is invalid");
394 }
395
396 validate_compute_sandbox_policy(&spec.sandbox)?;
397 validate_compute_artifact_policy(&spec.artifact_policy)?;
398 validate_compute_workload(&spec.workload)?;
399 validate_reducer_matches_workload(&spec.workload, &spec.reducer)?;
400 let mut seen = std::collections::BTreeSet::new();
401 for shard in &spec.shards {
402 validate_compute_shard_spec(&spec.workload, shard)?;
403 if !seen.insert(shard.shard_id.clone()) {
404 bail!("compute shard_id appears more than once");
405 }
406 }
407 Ok(())
408 }

431 pub fn validate_compute_artifact_policy(policy: &ComputeArtifactPolicy) -> Result<()> {
432 if policy.outputs.len() > MAX_COMPUTE_ARTIFACTS {
433 bail!("compute artifact policy has too many outputs");
434 }
435 if policy.max_total_bytes > MAX_COMPUTE_ARTIFACT_BYTES {
436 bail!("compute artifact policy max_total_bytes is invalid");
437 }
438 let mut seen = std::collections::BTreeSet::new();
439 let mut declared_total = 0u64;
440 for artifact in &policy.outputs {
441 validate_compute_artifact_path(&artifact.path)?;
442 if !seen.insert(artifact.path.clone()) {
443 bail!("compute artifact path appears more than once");
444 }
445 if let Some(max_bytes) = artifact.max_bytes {
446 if max_bytes == 0 || max_bytes > MAX_COMPUTE_ARTIFACT_BYTES {
447 bail!("compute artifact max_bytes is invalid");
448 }
449 declared_total = declared_total
450 .checked_add(max_bytes)
451 .ok_or_else(|| anyhow!("compute artifact byte limit overflow"))?;
452 }
453 if let Some(content_type) = &artifact.content_type
454 && (content_type.trim().is_empty()
455 || content_type.len() > crate::protocol::MAX_STORAGE_CONTENT_TYPE_BYTES)
456 {
457 bail!("compute artifact content_type is invalid");
458 }
459 }
460 if !policy.outputs.is_empty() && policy.max_total_bytes == 0 {
461 bail!("compute artifact policy with outputs requires max_total_bytes");
462 }
463 if policy.max_total_bytes > 0 && declared_total > policy.max_total_bytes {
464 bail!("compute artifact max_total_bytes is below declared output limits");
465 }
466 Ok(())
467 }

469 pub fn validate_compute_artifact_path(path: &str) -> Result<()> {
470 if path.trim().is_empty()
471 || path.len() > MAX_COMPUTE_ARTIFACT_PATH_BYTES
472 || path.starts_with('/')
473 || path.starts_with('\\')
474 || path.contains('\\')
475 || path.contains('\0')

478 bail!("compute artifact path is invalid");

486 .iter()
487 .all(|byte| byte.is_ascii_alphanumeric() || matches!(*byte, b'.' | b'-' | b'_'))
488 {

491 }

522 fn validate_compute_workload(workload: &ComputeWorkload) -> Result<()> {
523 if let ComputeWorkload::WasiPreview1 {
524 module_hex,
525 module_ref,
526 args,
527 env,
528 } = workload
529 {
530 match (module_hex, module_ref) {
531 (Some(module_hex), None) => {
532 let module = decode_hex_limited(module_hex, MAX_WASI_MODULE_BYTES, "wasi module")?;
533 if module.len() < 8 || &module[..4] != b"\0asm" {
534 bail!("wasi module is not a WebAssembly binary");
535 }
536 }
537 (None, Some(module_ref)) => validate_compute_module_ref(module_ref)?,
538 (Some(_), Some(_)) => {
539 bail!("wasi workload must not include both module_hex and module_ref")
540 }
541 (None, None) => bail!("wasi workload requires module_hex or module_ref"),
542 }
543 if args.len() > MAX_WASI_ARGS {
544 bail!("wasi workload has too many args");
545 }
546 for arg in args {
547 if arg.len() > MAX_WASI_ARG_BYTES || arg.contains('\0') {
548 bail!("wasi workload arg is invalid");
549 }
550 }
551 if env.len() > MAX_WASI_ENV_VARS {
552 bail!("wasi workload has too many env vars");
553 }
554 for (key, value) in env {
555 validate_wasi_env_var(key, value)?;
556 }
557 }
558 Ok(())
559 }

568 pub fn validate_compute_module_ref(module_ref: &ComputeModuleRef) -> Result<()> {
569 if module_ref.contract_id.trim().is_empty()
570 || module_ref.contract_id.len() > MAX_STORAGE_CONTRACT_ID_BYTES
571 {
572 bail!("compute module_ref contract_id is invalid");
573 }
574 if module_ref.host_url.trim().is_empty()
575 || module_ref.host_url.len() > MAX_WASI_MODULE_HOST_URL_BYTES
576 {
577 bail!("compute module_ref host_url is invalid");
578 }
579 let parsed = Url::parse(&module_ref.host_url)
580 .map_err(|_| anyhow!("compute module_ref host_url is invalid"))?;
581 if !matches!(parsed.scheme(), "http" | "https")
582 || !parsed.has_host()
583 || !parsed.username().is_empty()
584 || parsed.password().is_some()
585 || parsed.query().is_some()
586 || parsed.fragment().is_some()
587 || !matches!(parsed.path(), "" | "/")
588 {
589 bail!("compute module_ref host_url is invalid");
590 }
591 validate_compute_module_path(&module_ref.path)?;
592 validate_sha256_hex(&module_ref.sha256, "compute module_ref sha256")?;
593 validate_sha256_hex(&module_ref.merkle_root, "compute module_ref merkle_root")?;
594 if module_ref.size_bytes == 0 || module_ref.size_bytes > MAX_WASI_MODULE_REF_BYTES {
595 bail!("compute module_ref size_bytes is invalid");
596 }
597 Ok(())
598 }

643 pub fn validate_compute_shard_spec(
644 workload: &ComputeWorkload,
645 shard: &ComputeShardSpec,
646 ) -> Result<()> {
647 if shard.shard_id.trim().is_empty() || shard.shard_id.len() > MAX_COMPUTE_SHARD_ID_BYTES {
648 bail!("compute shard_id is invalid");
649 }
650 match (workload, &shard.input) {
651 (ComputeWorkload::IntegerMap { .. }, ComputeShardInput::Integers { values }) => {
652 if values.is_empty() || values.len() > MAX_COMPUTE_INTEGERS_PER_SHARD {
653 bail!("integer compute shard has an invalid value count");
654 }
655 if values
656 .iter()
657 .any(|value| value.abs() > MAX_COMPUTE_INTEGER_ABS)
658 {
659 bail!("integer compute shard value is outside the allowed range");
660 }
661 }
662 (ComputeWorkload::MonteCarloPi, ComputeShardInput::MonteCarlo { samples, .. }) => {
663 if *samples == 0 || *samples > MAX_MONTE_CARLO_SAMPLES_PER_SHARD {
664 bail!("monte_carlo_pi shard sample count is invalid");
665 }
666 }
667 (ComputeWorkload::WasiPreview1 { .. }, ComputeShardInput::Wasi { stdin_hex, files }) => {
668 let mut total_bytes = 0usize;
669 if let Some(stdin_hex) = stdin_hex {
670 total_bytes = total_bytes
671 .checked_add(
672 decode_hex_limited(stdin_hex, MAX_WASI_INPUT_FILE_BYTES, "wasi stdin")?
673 .len(),
674 )
675 .ok_or_else(|| anyhow!("wasi input size overflow"))?;
676 }
677 if files.len() > MAX_WASI_INPUT_FILES {
678 bail!("wasi shard has too many input files");
679 }
680 for (path, contents_hex) in files {
681 validate_compute_input_file_path(path)?;
682 total_bytes = total_bytes
683 .checked_add(
684 decode_hex_limited(
685 contents_hex,
686 MAX_WASI_INPUT_FILE_BYTES,
687 "wasi input file",
688 )?
689 .len(),
690 )
691 .ok_or_else(|| anyhow!("wasi input size overflow"))?;
692 }
693 if total_bytes > MAX_WASI_INPUT_TOTAL_BYTES {
694 bail!("wasi shard input total exceeds maximum size");
695 }
696 }
697 (ComputeWorkload::IntegerMap { .. }, _) => {
698 bail!("integer compute workload requires integer shards")
699 }
700 (ComputeWorkload::MonteCarloPi, _) => {
701 bail!("monte_carlo_pi workload requires monte_carlo shards")
702 }
703 (ComputeWorkload::WasiPreview1 { .. }, _) => bail!("wasi workload requires wasi shards"),
704 }
705 Ok(())
706 }

708 pub fn validate_compute_output(output: &ComputeShardOutput) -> Result<()> {
709 if output.shard_id.trim().is_empty() || output.shard_id.len() > MAX_COMPUTE_SHARD_ID_BYTES {
710 bail!("compute output shard_id is invalid");
711 }
712 let serialized = serde_json::to_vec(output)?;
713 if serialized.len() > MAX_COMPUTE_OUTPUT_BYTES {
714 bail!("compute shard output exceeds maximum size");
715 }
716 if output.success && output.output.is_none() {
717 bail!("successful compute shard output is missing output");
718 }
719 if !output.success
720 && output
721 .error
722 .as_deref()
723 .unwrap_or_default()
724 .trim()
725 .is_empty()
726 {
727 bail!("failed compute shard output is missing error");
728 }
729 if output.stdout_sample.as_deref().unwrap_or_default().len() > MAX_COMPUTE_STDIO_BYTES {
730 bail!("compute stdout sample exceeds maximum size");
731 }
732 if output.stderr_sample.as_deref().unwrap_or_default().len() > MAX_COMPUTE_STDIO_BYTES {
733 bail!("compute stderr sample exceeds maximum size");
734 }
735 if output.artifacts.len() > MAX_COMPUTE_ARTIFACTS {
736 bail!("compute shard output has too many artifacts");
737 }
738 for artifact in &output.artifacts {
739 validate_compute_artifact_path(&artifact.path)?;
740 validate_sha256_hex(&artifact.sha256, "compute artifact sha256")?;
741 if artifact.size_bytes > MAX_COMPUTE_ARTIFACT_BYTES {
742 bail!("compute artifact size exceeds maximum size");
743 }
744 if artifact.provider.trim().is_empty() {
745 bail!("compute artifact provider is invalid");
746 }
747 if artifact.uri.trim().is_empty() || artifact.uri.len() > 1024 {
748 bail!("compute artifact uri is invalid");
749 }
750 }
751 Ok(())
752 }

754 pub fn validate_compute_output_artifacts(
755 policy: &ComputeArtifactPolicy,
756 output: &ComputeShardOutput,
757 ) -> Result<()> {
758 validate_compute_artifact_policy(policy)?;
759 validate_compute_output(output)?;
760 let declared = policy
761 .outputs
762 .iter()
763 .map(|artifact| (artifact.path.clone(), artifact))
764 .collect::<BTreeMap<_, _>>();
765 let mut seen = std::collections::BTreeSet::new();
766 let mut total = 0u64;
767 for artifact in &output.artifacts {
768 let declared_artifact = declared
769 .get(&artifact.path)
770 .ok_or_else(|| anyhow!("compute artifact was not declared by policy"))?;
771 if !seen.insert(artifact.path.clone()) {
772 bail!("compute artifact appears more than once");
773 }
774 let max_bytes = declared_artifact
775 .max_bytes
776 .unwrap_or(policy.max_total_bytes);
777 if artifact.size_bytes > max_bytes {
778 bail!("compute artifact exceeds declared max_bytes");
779 }
780 if artifact.content_type != declared_artifact.content_type {
781 bail!("compute artifact content_type does not match policy");
782 }
783 total = total
784 .checked_add(artifact.size_bytes)
785 .ok_or_else(|| anyhow!("compute artifact total size overflow"))?;
786 }
787 if total > policy.max_total_bytes {
788 bail!("compute artifacts exceed policy max_total_bytes");
789 }
790 if output.success {
791 for artifact in &policy.outputs {
792 if artifact.required && !seen.contains(&artifact.path) {
793 bail!("successful compute output is missing a required artifact");
794 }
795 }
796 }
797 Ok(())
798 }

800 pub fn execute_compute_shard(
801 workload: &ComputeWorkload,
802 shard: &ComputeShardSpec,
803 ) -> Result<ComputeShardOutput> {
804 validate_compute_shard_spec(workload, shard)?;
805 let started = Instant::now();
806 let result = match (workload, &shard.input) {
807 (ComputeWorkload::IntegerMap { operation }, ComputeShardInput::Integers { values }) => {
808 execute_integer_shard(operation, values)
809 }
810 (ComputeWorkload::MonteCarloPi, ComputeShardInput::MonteCarlo { samples, seed }) => {
811 Ok(execute_monte_carlo_pi(*samples, *seed))
812 }
813 (ComputeWorkload::WasiPreview1 { .. }, _) => {
814 bail!("wasi compute shards require the sandbox runner")
815 }
816 _ => bail!("compute shard input does not match workload"),
817 };
818 let latency_ms = started.elapsed().as_millis().min(u128::from(u64::MAX)) as u64;
819 match result {
820 Ok(output) => Ok(ComputeShardOutput {
821 shard_id: shard.shard_id.clone(),
822 success: true,
823 latency_ms,
824 output: Some(output),
825 error: None,
826 artifacts: Vec::new(),
827 stdout_sample: None,
828 stderr_sample: None,
829 }),
830 Err(error) => Ok(ComputeShardOutput {
831 shard_id: shard.shard_id.clone(),
832 success: false,
833 latency_ms,
834 output: None,
835 error: Some(error.to_string()),
836 artifacts: Vec::new(),
837 stdout_sample: None,
838 stderr_sample: None,
839 }),
840 }
841 }

843 pub fn reduce_compute_outputs(
844 spec: &ComputeJobSpec,
845 outputs: &[ComputeShardOutput],
846 ) -> Result<Value> {
847 validate_compute_job_spec(spec)?;
848 if outputs.len() != spec.shards.len() {
849 bail!("compute reducer received the wrong number of shard outputs");
850 }
851 let expected = spec
852 .shards
853 .iter()
854 .map(|shard| shard.shard_id.clone())
855 .collect::<std::collections::BTreeSet<_>>();
856 let actual = outputs
857 .iter()
858 .map(|output| output.shard_id.clone())
859 .collect::<std::collections::BTreeSet<_>>();
860 if actual != expected {
861 bail!("compute reducer shard_id set does not match the job spec");

863 for output in outputs {
864 validate_compute_output(output)?;
865 if !output.success {
866 bail!("compute shard {} failed", output.shard_id);
867 }
868 }
869
870 match spec.reducer {

874 }
875 ComputeReducer::Count => {
876 reduce_u64_field(outputs, "count").map(|count| json!({ "count": count }))
877 }
878 ComputeReducer::MinMax => reduce_min_max(outputs),
879 ComputeReducer::MonteCarloPi => reduce_monte_carlo_pi(outputs),
880 ComputeReducer::ShardOutputs => Ok(json!({
881 "shards": outputs
882 .iter()
883 .map(|output| {
884 json!({
885 "shard_id": output.shard_id,
886 "output": output.output,
887 "artifacts": output.artifacts,
888 })
889 })
890 .collect::<Vec<_>>()
891 })),
892 }

950 fn execute_integer_shard(operation: &ComputeIntegerOperation, values: &[i64]) -> Result<Value> {
951 match operation {
952 ComputeIntegerOperation::Sum => {
953 let sum = checked_sum(values)?;
954 Ok(json!({ "sum": sum, "count": values.len() as u64 }))
955 }
956 ComputeIntegerOperation::SumSquares => {
957 let mut sum = 0i64;
958 for value in values {
959 let square = value
960 .checked_mul(*value)
961 .ok_or_else(|| anyhow!("integer square overflow"))?;
962 sum = sum
963 .checked_add(square)
964 .ok_or_else(|| anyhow!("sum_squares overflow"))?;
965 }
966 Ok(json!({ "sum_squares": sum, "count": values.len() as u64 }))
967 }
968 ComputeIntegerOperation::Count => Ok(json!({ "count": values.len() as u64 })),
969 ComputeIntegerOperation::MinMax => {
970 let min = values
971 .iter()
972 .min()
973 .ok_or_else(|| anyhow!("empty integer shard"))?;
974 let max = values
975 .iter()
976 .max()
977 .ok_or_else(|| anyhow!("empty integer shard"))?;
978 Ok(json!({ "min": min, "max": max, "count": values.len() as u64 }))
979 }
980 }
981 }

983 fn execute_monte_carlo_pi(samples: u64, seed: u64) -> Value {
984 let mut state = seed;
985 let mut inside = 0u64;
986 let scale = 1u128 << 53;
987 let radius_squared = scale * scale;
988 for _ in 0..samples {
989 let x = splitmix64_next(&mut state) >> 11;
990 let y = splitmix64_next(&mut state) >> 11;
991 let x = u128::from(x);
992 let y = u128::from(y);
993 if x * x + y * y <= radius_squared {
994 inside += 1;
995 }
996 }
997 json!({ "inside": inside, "samples": samples })
998 }
999
1000 fn splitmix64_next(state: &mut u64) -> u64 {
1001 *state = state.wrapping_add(0x9E37_79B9_7F4A_7C15);
1002 let mut z = *state;
1003 z = (z ^ (z >> 30)).wrapping_mul(0xBF58_476D_1CE4_E5B9);
1004 z = (z ^ (z >> 27)).wrapping_mul(0x94D0_49BB_1331_11EB);
1005 z ^ (z >> 31)
1006 }

1009 values.iter().try_fold(0i64, |sum, value| {
1010 sum.checked_add(*value)
1011 .ok_or_else(|| anyhow!("integer sum overflow"))
1012 })
1013 }

1037 fn reduce_min_max(outputs: &[ComputeShardOutput]) -> Result<Value> {
1038 let mut global_min: Option<i64> = None;
1039 let mut global_max: Option<i64> = None;
1040 let mut count = 0u64;
1041 for output in outputs {
1042 let value = output_value(output)?;
1043 let min = value
1044 .get("min")
1045 .and_then(Value::as_i64)
1046 .ok_or_else(|| anyhow!("compute output is missing min"))?;
1047 let max = value
1048 .get("max")
1049 .and_then(Value::as_i64)
1050 .ok_or_else(|| anyhow!("compute output is missing max"))?;
1051 let shard_count = value
1052 .get("count")
1053 .and_then(Value::as_u64)
1054 .ok_or_else(|| anyhow!("compute output is missing count"))?;
1055 global_min = Some(global_min.map(|existing| existing.min(min)).unwrap_or(min));
1056 global_max = Some(global_max.map(|existing| existing.max(max)).unwrap_or(max));
1057 count = count
1058 .checked_add(shard_count)
1059 .ok_or_else(|| anyhow!("compute count overflow"))?;
1060 }
1061 Ok(json!({
1062 "min": global_min.ok_or_else(|| anyhow!("missing min"))?,
1063 "max": global_max.ok_or_else(|| anyhow!("missing max"))?,
1064 "count": count,
1065 }))
1066 }
