1041 pub fn parse_amount(input: &str) -> Result<u64> {
1042 let trimmed = input.trim();
1043 if trimmed.is_empty() {
1044 bail!("amount cannot be empty");
1045 }
1046
1047 let negative = trimmed.starts_with('-');
1048 if negative {
1049 bail!("amount cannot be negative");
1050 }
1051
1052 let parts: Vec<&str> = trimmed.split('.').collect();
1053 if parts.len() > 2 {
1054 bail!("invalid amount format");
1055 }
1056
1057 let whole: u64 = parts[0].parse()?;
1058 let fractional = if let Some(fraction) = parts.get(1) {
1059 if fraction.len() > TOKEN_DECIMALS as usize {
1060 bail!("amount supports at most {TOKEN_DECIMALS} decimal places");
1061 }
1062
1063 let padded = format!("{fraction:0<width$}", width = TOKEN_DECIMALS as usize);
1064 padded.parse::<u64>()?
1065 } else {
1066 0
1067 };
1068
1069 whole
1070 .checked_mul(MICRO_HT)
1071 .and_then(|value| value.checked_add(fractional))
1072 .ok_or_else(|| anyhow!("amount overflow"))
1073 }
