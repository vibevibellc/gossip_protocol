64 def _dispatch(self, head_only: bool = False) -> None:
65 parsed = urlparse(self.path)
66 if parsed.path == "/health/ok":
67 self._handle_ok(parsed, head_only=head_only)
68 return
69 if parsed.path == "/health/head":
70 self._handle_head(parsed, head_only=head_only)
71 return
72 if parsed.path == "/health/post":
73 self._handle_post(parsed, head_only=head_only)
74 return
75 if parsed.path == "/health/degraded":
76 self._handle_degraded(parsed, head_only=head_only)
77 return
78 if parsed.path == "/health/slow":
79 self._handle_slow(parsed, head_only=head_only)
80 return
81 self.send_error(404, "not found")

89 def _json_response(
90 self,
91 status: int,
92 payload: dict,
93 *,
94 extra_headers: Dict[str, str] | None = None,
95 head_only: bool = False,
96 ) -> None:
97 body = json.dumps(payload).encode("utf-8")
98 self.send_response(status)
99 self.send_header("content-type", "application/json")
100 self.send_header("content-length", str(len(body)))
101 if extra_headers:
102 for name, value in extra_headers.items():
103 self.send_header(name, value)
104 self.end_headers()
105 if head_only:
106 return
107 try:
108 self.wfile.write(body)
109 except (BrokenPipeError, ConnectionResetError):
110 return

621 def verify_record(record: dict, scenario: Scenario, validator_count: int) -> None:
622 expected_success_count = validator_count if scenario.expected_success else 0
623 expected_failure_count = 0 if scenario.expected_success else validator_count
624
625 if record["success_count"] != expected_success_count:
626 raise RuntimeError(
627 f"{scenario.name} success_count mismatch: expected {expected_success_count}, "
628 f"saw {record['success_count']}"
629 )
630 if record["failure_count"] != expected_failure_count:
631 raise RuntimeError(
632 f"{scenario.name} failure_count mismatch: expected {expected_failure_count}, "
633 f"saw {record['failure_count']}"
634 )
635 if len(record["receipts"]) != validator_count:
636 raise RuntimeError(
637 f"{scenario.name} receipt count mismatch: expected {validator_count}, "
638 f"saw {len(record['receipts'])}"
639 )
640
641 for receipt in record["receipts"]:
642 body = receipt["body"]
643 if body["success"] != scenario.expected_success:
644 raise RuntimeError(
645 f"{scenario.name} receipt success mismatch for {body['executor']}: "
646 f"expected {scenario.expected_success}, saw {body['success']}"
647 )
648 if body["response_status"] != scenario.expected_receipt_status:
649 raise RuntimeError(
650 f"{scenario.name} response status mismatch: expected {scenario.expected_receipt_status}, "
651 f"saw {body['response_status']}"
652 )
653 if scenario.expect_error and not body["error"]:
654 raise RuntimeError(f"{scenario.name} expected an error but receipt had none")
655 if not scenario.expect_error and body["error"] is not None:
656 raise RuntimeError(f"{scenario.name} expected no error but saw {body['error']}")
657 if scenario.method == "head" and body["response_body_sample"] is not None:
658 raise RuntimeError(f"{scenario.name} expected HEAD receipts to have no body sample")

661 def main() -> int:

663 parser.add_argument("--keep-artifacts", action="store_true")
664 parser.add_argument("--cycles", type=int, default=1)

696 validator_addresses,
697 user_addresses,
698 total_waves,

702 node_specs = []
703 for index, wallet in enumerate(validator_wallets, start=1):
704 bind_port = 9500 + index
705 peers = [
706 f"http://127.0.0.1:{9500 + peer_index}"
707 for peer_index in range(1, VALIDATOR_COUNT + 1)
708 if peer_index != index
709 ]

712 wallet=wallet,

717 )

721
722 for process, spec in zip(processes, node_specs):
723 if not base.process_alive(process):
724 raise RuntimeError(f"{spec.name} exited early, see {spec.log_path}")
725
726 node_urls = [f"http://{spec.bind}" for spec in node_specs]
727 for url in node_urls:
728 base.wait_for_http(f"{url}/v1/control/ledger", 20)

731 scenarios_by_wave = build_scenarios(f"http://127.0.0.1:{health_port}")
732 if len(scenarios_by_wave) != WAVE_COUNT:
733 raise RuntimeError("scenario wave count does not match configured wave count")
734
735 job_results = []
736 start_time = time.time()
737 for cycle_index in range(1, args.cycles + 1):
738 for wave_offset, scenarios in enumerate(scenarios_by_wave, start=1):
739 wave_index = ((cycle_index - 1) * WAVE_COUNT) + wave_offset
740 submitted = []
741 with concurrent.futures.ThreadPoolExecutor(max_workers=USER_COUNT) as executor:
742 futures = []
743 for user_index, scenario in enumerate(scenarios):
744 wallet = user_wallets[user_index]
745 node_url = node_urls[(wave_index + user_index) % len(node_urls)]
746 request_url = f"http://127.0.0.1:{health_port}{scenario.path}"
747 futures.append(
748 executor.submit(submit_scenario, node_url, wallet, request_url, scenario)
749 )
750 for user_index, future in enumerate(futures):
751 tx = future.result()
752 submitted.append(
753 {
754 "user_wallet": str(user_wallets[user_index]),
755 "user_address": user_addresses[user_index],
756 "scenario": scenarios[user_index],
757 "tx": tx,

762 tx_hash = item["tx"]["hash"]
763 job = base.fetch_job_until_finalized(node_urls[0], tx_hash, 30)
764 record = job["finalized_record"]
765 verify_record(record, item["scenario"], VALIDATOR_COUNT)
766 job_results.append(
767 {
768 "cycle": cycle_index,
769 "wave": wave_index,
770 "user_address": item["user_address"],
771 "scenario": item["scenario"].name,
772 "tx_hash": tx_hash,
773 "success_count": record["success_count"],
774 "failure_count": record["failure_count"],
775 "requester_cost": record["requester_cost"],
776 }
777 )
778
779 base.verify_convergence(node_urls)
780 time.sleep(BLOCK_TIME_SECONDS + 0.25)
781
782 ledgers = base.verify_convergence(node_urls)
783 final_height = ledgers[0]["height"]
784 if final_height < total_waves:
785 raise RuntimeError(f"expected at least {total_waves} blocks, saw height {final_height}")
786 statuses = [base.fetch_status(url) for url in node_urls]
787 if any(status["validator_count"] != VALIDATOR_COUNT for status in statuses):
788 raise RuntimeError(f"unexpected validator count from status endpoint: {statuses}")
789 if any(status["height"] != final_height for status in statuses):
790 raise RuntimeError(f"status endpoint height mismatch after stress run: {statuses}")
791 if any(status["mempool_size"] != 0 for status in statuses):
792 raise RuntimeError(f"expected empty mempool after stress run: {statuses}")
793
794 for address in user_addresses:
795 accounts = [base.fetch_account(url, address) for url in node_urls]
796 if any(account["balance"] != 0 for account in accounts):
797 raise RuntimeError(f"user {address} balance mismatch after spend-down: {accounts}")
798
799 validator_balance_map = {}
800 expected_validator_balance = (USER_COUNT * total_waves * base.TOKEN) // VALIDATOR_COUNT
801 for address in validator_addresses:
802 accounts = [base.fetch_account(url, address) for url in node_urls]
803 if any(account["balance"] != expected_validator_balance for account in accounts):
804 raise RuntimeError(
805 f"validator {address} reward mismatch: expected {expected_validator_balance}, saw {accounts}"
806 )
807 validator_balance_map[address] = accounts[0]["balance"]
808
809 swap_quote = base.fetch_swap_quote(node_urls[0], user_addresses[0])
810 if swap_quote["quote"]["settlement_amount"] != 2 * base.TOKEN:
811 raise RuntimeError("swap quote math regressed under stress")
812
813 elapsed_ms = int((time.time() - start_time) * 1000)
814 success_jobs = sum(1 for result in job_results if result["success_count"] == VALIDATOR_COUNT)
815 failed_jobs = len(job_results) - success_jobs
816

819 "chain_id": genesis["chain_id"],
820 "validator_count": VALIDATOR_COUNT,

823 "cycles": args.cycles,
824 "jobs_submitted": len(job_results),
825 "success_jobs": success_jobs,
826 "failed_jobs": failed_jobs,
827 "final_height": final_height,
828 "elapsed_ms": elapsed_ms,
829 "validator_balances_micro_ht": validator_balance_map,
830 "job_results": job_results,
831 "swap_quote": swap_quote,
832 "node_urls": node_urls,
833 "statuses": statuses,
834 "logs": {spec.name: str(spec.log_path) for spec in node_specs},
835 }
