29 pub async fn execute_health_check(spec: &HealthCheckSpec) -> HealthExecution {
30 match execute_inner(spec).await {
31 Ok(outcome) => outcome,
32 Err(error) => HealthExecution {
33 response_status: None,
34 latency_ms: 0,
35 success: false,
36 assertion_results: Vec::new(),
37 response_headers: BTreeMap::new(),
38 response_body_sample: None,
39 error: Some(error.to_string()),
40 },
41 }

44 async fn execute_inner(spec: &HealthCheckSpec) -> Result<HealthExecution> {
45 validate_health_check_spec(spec)?;
46
47 let mut url = Url::parse(&spec.url)?;
48 if url.scheme() != "https" && !spec.allow_insecure_http {
49 bail!("health checks require https unless allow_insecure_http is true");
50 }
51
52 {
53 let mut pairs = url.query_pairs_mut();
54 for (key, value) in &spec.query {
55 pairs.append_pair(key, value);
56 }
57 }
58
59 if !spec.allow_private_targets {
60 enforce_public_target_policy(&url).await?;
61 }
62
63 let client = reqwest::Client::builder()
64 .timeout(std::time::Duration::from_millis(spec.timeout_ms))
65 .build()?;
66
67 let mut headers = HeaderMap::new();
68 for (key, value) in &spec.headers {
69 headers.insert(
70 HeaderName::from_bytes(key.as_bytes())?,
71 HeaderValue::from_str(value)?,
72 );
73 }
74
75 let start = Instant::now();
76 let request = match spec.method {
77 HealthHttpMethod::Get => client.get(url),
78 HealthHttpMethod::Head => client.head(url),
79 HealthHttpMethod::Post => client.post(url),
80 }
81 .headers(headers);
82
83 let request = if matches!(spec.method, HealthHttpMethod::Post) {
84 if let Some(body) = &spec.body_json {
85 request.json(body)
86 } else {
87 request
88 }
89 } else {
90 request
91 };
92
93 let response = request.send().await?;
94 let latency_ms = start.elapsed().as_millis() as u64;
95 let status = response.status().as_u16();
96 let response_headers = simplify_headers(response.headers());
97 let body = if matches!(spec.method, HealthHttpMethod::Head) {
98 None
99 } else {
100 Some(response.text().await?)
101 };
102
103 let body_sample = body.as_deref().map(truncate_body);
104 let parsed_json = body
105 .as_deref()
106 .and_then(|text| serde_json::from_str::<Value>(text).ok());
107 let assertion_results = evaluate_assertions(
108 spec,
109 &response_headers,
110 body.as_deref(),
111 parsed_json.as_ref(),
112 );
113
114 let expected_status_ok = spec
115 .expected_status
116 .map(|expected| expected == status)
117 .unwrap_or(true);
118 let all_assertions_ok = assertion_results.iter().all(|result| result.passed);
119 let success = expected_status_ok && all_assertions_ok;
120
121 Ok(HealthExecution {
122 response_status: Some(status),
123 latency_ms,
124 success,
125 assertion_results,
126 response_headers,
127 response_body_sample: body_sample,
128 error: None,
129 })
130 }

132 async fn enforce_public_target_policy(url: &Url) -> Result<()> {
133 let host = url
134 .host_str()
135 .ok_or_else(|| anyhow!("health check url must include a host"))?;
136 if is_disallowed_hostname(host) {
137 bail!("health check target host is not allowed");
138 }
139
140 let port = url
141 .port_or_known_default()
142 .ok_or_else(|| anyhow!("health check target port is invalid"))?;
143 let resolved: Vec<_> = lookup_host((host, port)).await?.collect();
144 if resolved.is_empty() {
145 bail!("health check target did not resolve to any addresses");
146 }
147 if resolved
148 .iter()
149 .any(|socket_addr| is_disallowed_ip(socket_addr.ip()))
150 {
151 bail!("health check target resolved to a private or local address");
152 }
153 Ok(())
154 }

156 fn is_disallowed_hostname(host: &str) -> bool {
157 host.eq_ignore_ascii_case("localhost")
158 || host.ends_with(".local")
159 || host.ends_with(".internal")
160 || host.ends_with(".localhost")
161 }

163 fn is_disallowed_ip(ip: IpAddr) -> bool {
164 match ip {
165 IpAddr::V4(ip) => {
166 ip.is_private()
167 || ip.is_loopback()
168 || ip.is_link_local()
169 || ip.is_multicast()
170 || ip.is_unspecified()
171 || ip.is_broadcast()
172 || is_documentation_ipv4(ip)
173 }
174 IpAddr::V6(ip) => {
175 ip.is_loopback()
176 || ip.is_multicast()
177 || ip.is_unspecified()
178 || ip.is_unique_local()
179 || ip.is_unicast_link_local()
180 || is_documentation_ipv6(ip)
181 || ip
182 .to_ipv4_mapped()
183 .map(|ipv4| is_disallowed_ip(IpAddr::V4(ipv4)))
184 .unwrap_or(false)
185 }
186 }
187 }

189 fn is_documentation_ipv4(ip: Ipv4Addr) -> bool {
190 let octets = ip.octets();
191 matches!(
192 octets,
193 [192, 0, 2, _] | [198, 51, 100, _] | [203, 0, 113, _]
194 )
195 }

197 fn is_documentation_ipv6(ip: Ipv6Addr) -> bool {
198 let segments = ip.segments();
199 segments[0] == 0x2001 && segments[1] == 0x0db8
200 }

214 fn truncate_body(body: &str) -> String {
215 let bytes = body.as_bytes();
216 if bytes.len() <= MAX_BODY_SAMPLE_BYTES {
217 return body.to_string();
218 }
219
220 String::from_utf8_lossy(&bytes[..MAX_BODY_SAMPLE_BYTES]).to_string()
221 }

223 fn evaluate_assertions(
224 spec: &HealthCheckSpec,
225 headers: &BTreeMap<String, String>,
226 body: Option<&str>,
227 json: Option<&Value>,
228 ) -> Vec<AssertionResult> {
229 spec.assertions
230 .iter()
231 .map(|assertion| match assertion {
232 ResponseAssertion::JsonFieldExists { path } => {
233 let found = json.and_then(|value| find_json_path(value, path)).is_some();
234 AssertionResult {
235 assertion: assertion.clone(),
236 passed: found,
237 detail: if found {
238 format!("json path {path} exists")
239 } else {
240 format!("json path {path} is missing")
241 },
242 }
243 }
244 ResponseAssertion::JsonFieldEquals { path, value } => {
245 let observed = json.and_then(|candidate| find_json_path(candidate, path));
246 let passed = observed
247 .map(|candidate| candidate == value)
248 .unwrap_or(false);
249 AssertionResult {
250 assertion: assertion.clone(),
251 passed,
252 detail: if let Some(observed) = observed {
253 format!("expected {value} and observed {observed}")
254 } else {
255 format!("json path {path} is missing")
256 },
257 }
258 }
259 ResponseAssertion::HeaderEquals { name, value } => {
260 let observed = headers
261 .iter()
262 .find(|(key, _)| key.eq_ignore_ascii_case(name))
263 .map(|(_, value)| value.as_str());
264 let passed = observed
265 .map(|candidate| candidate == value)
266 .unwrap_or(false);
267 AssertionResult {
268 assertion: assertion.clone(),
269 passed,
270 detail: if let Some(observed) = observed {
271 format!("expected header {name}={value} and observed {observed}")
272 } else {
273 format!("header {name} is missing")
274 },
275 }
276 }
277 ResponseAssertion::BodyContains { text } => {
278 let passed = body
279 .map(|candidate| candidate.contains(text))
280 .unwrap_or(false);
281 AssertionResult {
282 assertion: assertion.clone(),
283 passed,
284 detail: if passed {
285 format!("body contains {text}")
286 } else {
287 format!("body does not contain {text}")
288 },
289 }
290 }
291 })
292 .collect()
293 }

295 fn find_json_path<'a>(value: &'a Value, path: &str) -> Option<&'a Value> {
296 let mut current = value;
297 for segment in path.split('.') {
298 current = match current {
299 Value::Object(map) => map.get(segment)?,
300 Value::Array(items) => {
301 let index: usize = segment.parse().ok()?;
302 items.get(index)?
303 }
304 _ => return None,
305 };
306 }
307 Some(current)
308 }
